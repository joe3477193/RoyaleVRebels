13  package net.sf.oval.internal.util;
14  
15  import static net.sf.oval.Validator.*;
16  
17  import java.io.ObjectInputStream;
18  import java.io.ObjectOutputStream;
19  import java.io.Serializable;
20  import java.util.Collection;
21  import java.util.Iterator;
22  import java.util.Map;
23  import java.util.Set;
24  
25  /**
26   * @author Sebastian Thomschke
27   */
28  public final class IdentitySet<E> implements Set<E>, Serializable {
29      private static final long serialVersionUID = 1L;
30  
31      private transient Map<Integer, E> map;
32  
33      /**
34       * Constructs a new, empty <tt>IdentitySet</tt>; the backing <tt>Map</tt> instance has
35       * default initial capacity (16) and load factor (0.75).
36       */
37      public IdentitySet() {
38          map = getCollectionFactory().createMap();
39      }
40  
41      /**
42       * Constructs a new, empty <tt>IdentitySet</tt>; the backing <tt>Map</tt> instance has
43       * the given initial capacity and the default load factor (0.75).
44       */
45      public IdentitySet(final int initialCapacity) {
46          map = getCollectionFactory().createMap(initialCapacity);
47      }
48  
49      public boolean add(final E o) {
50          final int hash = System.identityHashCode(o);
51          return map.put(hash, o) == null;
52      }
53  
54      public boolean addAll(final Collection<? extends E> c) {
55          int count = 0;
56          for (final E e : c)
57              if (add(e)) {
58                  count++;
59              }
60          return count > 0;
61      }
62  
63      public void clear() {
64          map.clear();
65      }
66  
67      public boolean contains(final Object o) {
68          final int hash = System.identityHashCode(o);
69          return map.containsKey(hash);
70      }
71  
72      public boolean containsAll(final Collection<?> c) {
73          throw new UnsupportedOperationException();
74      }
75  
76      public boolean isEmpty() {
77          return map.isEmpty();
78      }
79  
80      public Iterator<E> iterator() {
81          return map.values().iterator();
82      }
83  
84      /**
85       * Reads the <tt>IdentitySet</tt> instance from a stream.
86       */
87      @SuppressWarnings("unchecked")
88      private void readObject(final ObjectInputStream ois) throws java.io.IOException, ClassNotFoundException {
89          // materialize any hidden serialization magic
90          ois.defaultReadObject();
91  
92          // materialize the size
93          final int size = ois.readInt();
94  
95          // materialize the elements
96          map = getCollectionFactory().createMap(size);
97          for (int i = 0; i < size; i++) {
98              final E o = (E) ois.readObject();
99              final int hash = System.identityHashCode(o);
100             map.put(hash, o);
101         }
102     }
103 
104     public boolean remove(final Object o) {
105         final int hash = System.identityHashCode(o);
106         return map.remove(hash) != null;
107     }
108 
109     public boolean removeAll(final Collection<?> c) {
110         boolean modified = false;
111         for (final Object e : c)
112             if (remove(e)) {
113                 modified = true;
114             }
115         return modified;
116     }
117 
118     public boolean retainAll(final Collection<?> c) {
119         throw new UnsupportedOperationException();
120     }
121 
122     public int size() {
123         return map.size();
124     }
125 
126     public Object[] toArray() {
127         return map.values().toArray();
128     }
129 
130     public <T> T[] toArray(final T[] a) {
131         return map.values().toArray(a);
132     }
133 
134     /**
135      * Writes state of this <tt>IdentitySet</tt> instance to a stream.
136      */
137     private void writeObject(final ObjectOutputStream oos) throws java.io.IOException {
138         // serialize any hidden serialization magic
139         oos.defaultWriteObject();
140 
141         // serialize the set's size
142         oos.writeInt(map.size());
143 
144         // serialize the set's elements
145         for (final E e : map.values()) {
146             oos.writeObject(e);
147         }
148     }
149 }
