13  package net.sf.oval;
14  
15  import static net.sf.oval.Validator.*;
16  
17  import java.util.Map;
18  
19  import net.sf.oval.expression.ExpressionLanguage;
20  
21  /**
22   * Partial implementation of exclusion classes.
23   *
24   * @author Sebastian Thomschke
25   */
26  public abstract class AbstractCheckExclusion implements CheckExclusion {
27      private static final long serialVersionUID = 1L;
28  
29      private String[] profiles;
30  
31      private String when;
32      private String whenFormula;
33      private String whenLang;
34  
35      public Map<String, String> getMessageVariables() {
36          return null;
37      }
38  
39      public String[] getProfiles() {
40          return profiles;
41      }
42  
43      public String getWhen() {
44          return whenLang + ":" + when;
45      }
46  
47      public boolean isActive(final Object validatedObject, final Object valueToValidate, final Validator validator) {
48          if (when == null)
49              return true;
50  
51          final Map<String, Object> values = getCollectionFactory().createMap();
52          values.put("_value", valueToValidate);
53          values.put("_this", validatedObject);
54  
55          final ExpressionLanguage el = validator.getExpressionLanguageRegistry().getExpressionLanguage(whenLang);
56          return el.evaluateAsBoolean(whenFormula, values);
57      }
58  
59      public void setProfiles(final String... profiles) {
60          this.profiles = profiles;
61      }
62  
63      public void setWhen(final String when) {
64          if (when == null || when.length() == 0) {
65              this.when = null;
66              whenFormula = null;
67              whenLang = null;
68          } else {
69              this.when = when;
70              final String[] parts = when.split(":", 2);
71              if (parts.length == 0)
72                  throw new IllegalArgumentException("[when] is missing the scripting language declaration");
73              whenLang = parts[0];
74              whenFormula = parts[1];
75          }
76      }
77  }
