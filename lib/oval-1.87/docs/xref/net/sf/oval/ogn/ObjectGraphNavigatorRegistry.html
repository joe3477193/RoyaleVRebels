13  package net.sf.oval.ogn;
14  
15  import java.util.Map;
16  
17  import net.sf.oval.Validator;
18  import net.sf.oval.exception.ObjectGraphNavigatorNotAvailableException;
19  import net.sf.oval.internal.Log;
20  import net.sf.oval.internal.util.Assert;
21  import net.sf.oval.internal.util.ReflectionUtils;
22  
23  /**
24   * @author Sebastian Thomschke
25   *
26   */
27  public class ObjectGraphNavigatorRegistry {
28      private static final Log LOG = Log.getLog(ObjectGraphNavigatorRegistry.class);
29  
30      private final Map<String, ObjectGraphNavigator> cache = Validator.getCollectionFactory().createMap(2);
31  
32      private ObjectGraphNavigator _initializeDefaultOGN(final String id) {
33          // JXPath support
34          if ("jxpath".equals(id) && ReflectionUtils.isClassPresent("org.apache.commons.jxpath.JXPathContext"))
35              return registerObjectGraphNavigator("jxpath", new ObjectGraphNavigatorJXPathImpl());
36  
37          if ("".equals(id))
38              return registerObjectGraphNavigator("", new ObjectGraphNavigatorDefaultImpl());
39          return null;
40      }
41  
42      public ObjectGraphNavigator getObjectGraphNavigator(final String id) {
43          Assert.argumentNotNull("id", id);
44  
45          ObjectGraphNavigator ogn = cache.get(id);
46  
47          if (ogn == null)
48              ogn = _initializeDefaultOGN(id);
49  
50          if (ogn == null)
51              throw new ObjectGraphNavigatorNotAvailableException(id);
52  
53          return ogn;
54      }
55  
56      public ObjectGraphNavigator registerObjectGraphNavigator(final String id, final ObjectGraphNavigator ogn) throws IllegalArgumentException {
57          Assert.argumentNotNull("id", id);
58          Assert.argumentNotNull("ogn", ogn);
59  
60          LOG.info("Object Graph Navigator '{1}' registered: {2}", id, ogn);
61  
62          cache.put(id, ogn);
63          return ogn;
64      }
65  }
