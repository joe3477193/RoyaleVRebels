36  /**
37   * JXPath {@link "http://commons.apache.org/jxpath/"} based object graph navigator implementation.
38   * 
39   * @author Sebastian Thomschke
40   */
41  public class ObjectGraphNavigatorJXPathImpl implements ObjectGraphNavigator {
42      protected static final class BeanPointerEx extends BeanPointer {
43          private static final long serialVersionUID = 1L;
44  
45          private final JXPathBeanInfo beanInfo;
46  
47          public BeanPointerEx(final NodePointer parent, final QName name, final Object bean, final JXPathBeanInfo beanInfo) {
48              super(parent, name, bean, beanInfo);
49              this.beanInfo = beanInfo;
50          }
51  
52          public BeanPointerEx(final QName name, final Object bean, final JXPathBeanInfo beanInfo, final Locale locale) {
53              super(name, bean, beanInfo, locale);
54              this.beanInfo = beanInfo;
55          }
56  
57          @Override
58          public boolean equals(final Object obj) {
59              if (this == obj)
60                  return true;
61              if (!super.equals(obj))
62                  return false;
63              if (getClass() != obj.getClass())
64                  return false;
65              final BeanPointerEx other = (BeanPointerEx) obj;
66              if (beanInfo == null) {
67                  if (other.beanInfo != null)
68                      return false;
69              } else if (!beanInfo.equals(other.beanInfo))
70                  return false;
71              return true;
72          }
73  
74          @Override
75          public boolean isValidProperty(final QName name) {
76              if (!super.isValidProperty(name))
77                  return false;
78  
79              // JXPath's default implementation returns true, even if the given property does not exit
80              if (beanInfo.getPropertyDescriptor(name.getName()) == null)
81                  throw new JXPathNotFoundException("No pointer for xpath: " + toString() + "/" + name);
82  
83              return true;
84          }
85      }
86  
87      protected static final class BeanPointerFactoryEx extends BeanPointerFactory {
88          @Override
89          public NodePointer createNodePointer(final NodePointer parent, final QName name, final Object bean) {
90              if (bean == null)
91                  return new NullPointer(parent, name);
92  
93              final JXPathBeanInfo bi = JXPathIntrospector.getBeanInfo(bean.getClass());
94              return new BeanPointerEx(parent, name, bean, bi);
95          }
96  
97          @Override
98          public NodePointer createNodePointer(final QName name, final Object bean, final Locale locale) {
99              final JXPathBeanInfo bi = JXPathIntrospector.getBeanInfo(bean.getClass());
100             return new BeanPointerEx(name, bean, bi, locale);
101         }
102 
103         @Override
104         public int getOrder() {
105             return BeanPointerFactory.BEAN_POINTER_FACTORY_ORDER - 1;
106         }
107     }
108 
109     static {
110         /*
111          * JXPath currently does not distinguish between invalid object graph paths, e.g. by referencing a non-existing property on a Java Bean,
112          * and incomplete object graph paths because of null-values.
113          * In both cases a JXPathNotFoundException is thrown if JXPathContext.lenient is <code>false</code>, and in both cases a NullPropertyPointer is returned if
114          * JXPathContext.lenient is <code>true</code>.
115          *
116          * Therefore we install a patched BeanPointerFactory that checks the existence of properties and throws a JXPathNotFoundException if it does not exist, no matter
117          * to which setting JXPathContext.lenient is set.
118          */
119         JXPathContextReferenceImpl.addNodePointerFactory(new BeanPointerFactoryEx());
120     }
121 
122     public ObjectGraphNavigationResult navigateTo(final Object root, final String xpath) throws InvalidConfigurationException {
123         Assert.argumentNotNull("root", root);
124         Assert.argumentNotNull("xpath", xpath);
125 
126         try {
127             final JXPathContext ctx = JXPathContext.newContext(root);
128             ctx.setLenient(true); // do not throw an exception if object graph is incomplete, e.g. contains null-values
129 
130             Pointer pointer = ctx.getPointer(xpath);
131 
132             // no match found or invalid xpath
133             if (pointer instanceof NullPropertyPointer || pointer instanceof NullPointer)
134                 return null;
135 
136             if (pointer instanceof BeanPointer) {
137                 final Pointer parent = ((BeanPointer) pointer).getImmediateParentPointer();
138                 if (parent instanceof PropertyPointer) {
139                     pointer = parent;
140                 }
141             }
142 
143             if (pointer instanceof PropertyPointer) {
144                 final PropertyPointer pp = (PropertyPointer) pointer;
145                 final Class<?> beanClass = pp.getBean().getClass();
146                 AccessibleObject accessor = ReflectionUtils.getField(beanClass, pp.getPropertyName());
147                 if (accessor == null) {
148                     accessor = ReflectionUtils.getGetter(beanClass, pp.getPropertyName());
149                 }
150                 return new ObjectGraphNavigationResult(root, xpath, pp.getBean(), accessor, pointer.getValue());
151             }
152 
153             throw new InvalidConfigurationException("Don't know how to handle pointer [" + pointer + "] of type [" + pointer.getClass().getName()
154                     + "] for xpath [" + xpath + "]");
155         } catch (final JXPathNotFoundException ex) {
156             // thrown if the xpath is invalid
157             throw new InvalidConfigurationException(ex);
158         }
159     }
