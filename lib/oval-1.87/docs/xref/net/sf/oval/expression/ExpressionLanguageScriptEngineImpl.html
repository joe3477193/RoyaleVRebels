13  package net.sf.oval.expression;
14  
15  import java.util.List;
16  import java.util.Map;
17  import java.util.Map.Entry;
18  
19  import javax.script.Bindings;
20  import javax.script.Compilable;
21  import javax.script.CompiledScript;
22  import javax.script.ScriptEngine;
23  import javax.script.ScriptEngineFactory;
24  import javax.script.ScriptEngineManager;
25  import javax.script.ScriptException;
26  
27  import net.sf.oval.Validator;
28  import net.sf.oval.exception.ExpressionEvaluationException;
29  import net.sf.oval.internal.Log;
30  import net.sf.oval.internal.util.ObjectCache;
31  
32  /**
33   * JSR223 Support
34   *
35   * @author Sebastian Thomschke
36   */
37  public class ExpressionLanguageScriptEngineImpl extends AbstractExpressionLanguage {
38      private static final Log LOG = Log.getLog(ExpressionLanguageScriptEngineImpl.class);
39  
40      private static final ScriptEngineManager FACTORY = new ScriptEngineManager();
41  
42      static {
43          final List<Object> languages = Validator.getCollectionFactory().createList();
44          for (final ScriptEngineFactory ef : FACTORY.getEngineFactories()) {
45              languages.add(ef.getNames());
46          }
47          LOG.info("Available ScriptEngine language names: {1}", languages);
48      }
49  
50      public static ExpressionLanguageScriptEngineImpl get(final String languageId) {
51          final ScriptEngine engine = FACTORY.getEngineByName(languageId);
52          return engine == null ? null : new ExpressionLanguageScriptEngineImpl(engine);
53      }
54  
55      private final Compilable compilable;
56      private final ScriptEngine engine;
57      private final ObjectCache<String, CompiledScript> compiledCache;
58  
59      private ExpressionLanguageScriptEngineImpl(final ScriptEngine engine) {
60          this.engine = engine;
61          if (engine instanceof Compilable) {
62              compilable = (Compilable) engine;
63              compiledCache = new ObjectCache<String, CompiledScript>();
64          } else {
65              compilable = null;
66              compiledCache = null;
67          }
68      }
69  
70      public Object evaluate(final String expression, final Map<String, ?> values) throws ExpressionEvaluationException {
71          LOG.debug("Evaluating JavaScript expression: {1}", expression);
72          try {
73              final Bindings scope = engine.createBindings();
74              for (final Entry<String, ?> entry : values.entrySet()) {
75                  scope.put(entry.getKey(), entry.getValue());
76              }
77  
78              if (compilable != null) {
79                  CompiledScript compiled = compiledCache.get(expression);
80                  if (compiled == null) {
81                      compiled = compilable.compile(expression);
82                      compiledCache.put(expression, compiled);
83                  }
84                  return compiled.eval(scope);
85              }
86              return engine.eval(expression, scope);
87          } catch (final ScriptException ex) {
88              throw new ExpressionEvaluationException("Evaluating JavaScript expression failed: " + expression, ex);
89          }
90      }
91  }
