13  package net.sf.oval.expression;
14  
15  import java.util.Map;
16  
17  import net.sf.oval.Validator;
18  import net.sf.oval.exception.ExpressionLanguageNotAvailableException;
19  import net.sf.oval.internal.Log;
20  import net.sf.oval.internal.util.Assert;
21  import net.sf.oval.internal.util.ReflectionUtils;
22  
23  /**
24   * @author Sebastian Thomschke
25   */
26  public class ExpressionLanguageRegistry {
27      private static final Log LOG = Log.getLog(ExpressionLanguageRegistry.class);
28  
29      private final Map<String, ExpressionLanguage> elcache = Validator.getCollectionFactory().createMap(4);
30  
31      private ExpressionLanguage _initializeDefaultEL(final String languageId) {
32          // JavaScript support
33          if (("javascript".equals(languageId) || "js".equals(languageId)) && ReflectionUtils.isClassPresent("org.mozilla.javascript.Context"))
34              return registerExpressionLanguage("js", registerExpressionLanguage("javascript", new ExpressionLanguageJavaScriptImpl()));
35  
36          // Groovy support
37          if ("groovy".equals(languageId) && ReflectionUtils.isClassPresent("groovy.lang.Binding"))
38              return registerExpressionLanguage("groovy", new ExpressionLanguageGroovyImpl());
39  
40          // BeanShell support
41          if (("beanshell".equals(languageId) || "bsh".equals(languageId)) && ReflectionUtils.isClassPresent("bsh.Interpreter"))
42              return registerExpressionLanguage("beanshell", registerExpressionLanguage("bsh", new ExpressionLanguageBeanShellImpl()));
43  
44          // OGNL support
45          if ("ognl".equals(languageId) && ReflectionUtils.isClassPresent("ognl.Ognl"))
46              return registerExpressionLanguage("ognl", new ExpressionLanguageOGNLImpl());
47  
48          // MVEL2 support
49          if ("mvel".equals(languageId) && ReflectionUtils.isClassPresent("org.mvel2.MVEL"))
50              return registerExpressionLanguage("mvel", new ExpressionLanguageMVELImpl());
51  
52          // JRuby support
53          else if (("jruby".equals(languageId) || "ruby".equals(languageId)) && ReflectionUtils.isClassPresent("org.jruby.Ruby"))
54              return registerExpressionLanguage("jruby", registerExpressionLanguage("ruby", new ExpressionLanguageJRubyImpl()));
55  
56          // JEXL2 support
57          if ("jexl".equals(languageId) && ReflectionUtils.isClassPresent("org.apache.commons.jexl2.JexlEngine"))
58              return registerExpressionLanguage("jexl", new ExpressionLanguageJEXLImpl());
59  
60          // scripting support via JSR223
61          if (ReflectionUtils.isClassPresent("javax.script.ScriptEngineManager")) {
62              final ExpressionLanguage el = ExpressionLanguageScriptEngineImpl.get(languageId);
63              if (el != null)
64                  return registerExpressionLanguage(languageId, el);
65          }
66  
67          return null;
68      }
69  
70      /**
71       *
72       * @param languageId the id of the language, cannot be null
73       *
74       * @throws IllegalArgumentException if <code>languageName == null</code>
75       * @throws ExpressionLanguageNotAvailableException
76       */
77      public ExpressionLanguage getExpressionLanguage(final String languageId) throws IllegalArgumentException, ExpressionLanguageNotAvailableException {
78          Assert.argumentNotNull("languageId", languageId);
79  
80          ExpressionLanguage el = elcache.get(languageId);
81  
82          if (el == null)
83              el = _initializeDefaultEL(languageId);
84  
85          if (el == null)
86              throw new ExpressionLanguageNotAvailableException(languageId);
87  
88          return el;
89      }
90  
91      /**
92       *
93       * @param languageId the expression language identifier
94       * @param impl the expression language implementation
95       * @throws IllegalArgumentException if <code>languageId == null || expressionLanguage == null</code>
96       */
97      public ExpressionLanguage registerExpressionLanguage(final String languageId, final ExpressionLanguage impl) throws IllegalArgumentException {
98          Assert.argumentNotNull("languageId", languageId);
99          Assert.argumentNotNull("impl", impl);
100 
101         LOG.info("Expression language '{1}' registered: {2}", languageId, impl);
102         elcache.put(languageId, impl);
103         return impl;
104     }
105 }
