13  package net.sf.oval.expression;
14  
15  import java.util.Map;
16  
17  import net.sf.oval.exception.ExpressionEvaluationException;
18  import net.sf.oval.internal.Log;
19  import net.sf.oval.internal.util.ObjectCache;
20  
21  import org.mvel2.MVEL;
22  
23  /**
24   * @author Sebastian Thomschke
25   */
26  public class ExpressionLanguageMVELImpl extends AbstractExpressionLanguage {
27      private static final Log LOG = Log.getLog(ExpressionLanguageMVELImpl.class);
28  
29      private final ObjectCache<String, Object> expressionCache = new ObjectCache<String, Object>();
30  
31      public Object evaluate(final String expression, final Map<String, ?> values) throws ExpressionEvaluationException {
32          LOG.debug("Evaluating MVEL expression: {1}", expression);
33          try {
34              Object expr = expressionCache.get(expression);
35              if (expr == null) {
36                  expr = MVEL.compileExpression(expression);
37                  expressionCache.put(expression, expr);
38              }
39              return MVEL.executeExpression(expr, values);
40          } catch (final Exception ex) {
41              throw new ExpressionEvaluationException("Evaluating MVEL expression failed: " + expression, ex);
42          }
43      }
44  }
