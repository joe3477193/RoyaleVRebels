13  package net.sf.oval.expression;
14  
15  import java.util.Map;
16  import java.util.Map.Entry;
17  
18  import net.sf.oval.exception.ExpressionEvaluationException;
19  import net.sf.oval.internal.Log;
20  import bsh.EvalError;
21  import bsh.Interpreter;
22  
23  /**
24   * @author Sebastian Thomschke
25   */
26  public class ExpressionLanguageBeanShellImpl extends AbstractExpressionLanguage {
27      private static final Log LOG = Log.getLog(ExpressionLanguageBeanShellImpl.class);
28  
29      public Object evaluate(final String expression, final Map<String, ?> values) throws ExpressionEvaluationException {
30          LOG.debug("Evaluating BeanShell expression: {1}", expression);
31          try {
32              final Interpreter interpreter = new Interpreter();
33              interpreter.eval("setAccessibility(true)"); // turn off access restrictions
34              for (final Entry<String, ?> entry : values.entrySet()) {
35                  interpreter.set(entry.getKey(), entry.getValue());
36              }
37              return interpreter.eval(expression);
38          } catch (final EvalError ex) {
39              throw new ExpressionEvaluationException("Evaluating BeanShell expression failed: " + expression, ex);
40          }
41      }
42  }
