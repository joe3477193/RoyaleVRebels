13  package net.sf.oval.expression;
14  
15  import groovy.lang.Binding;
16  import groovy.lang.GroovyShell;
17  import groovy.lang.Script;
18  
19  import java.util.Map;
20  import java.util.Map.Entry;
21  
22  import net.sf.oval.exception.ExpressionEvaluationException;
23  import net.sf.oval.internal.Log;
24  import net.sf.oval.internal.util.ObjectCache;
25  import net.sf.oval.internal.util.ThreadLocalObjectCache;
26  
27  /**
28   * @author Sebastian Thomschke
29   */
30  public class ExpressionLanguageGroovyImpl extends AbstractExpressionLanguage {
31      private static final Log LOG = Log.getLog(ExpressionLanguageGroovyImpl.class);
32  
33      private static final GroovyShell GROOVY_SHELL = new GroovyShell();
34  
35      private final ThreadLocalObjectCache<String, Script> threadScriptCache = new ThreadLocalObjectCache<String, Script>();
36  
37      public Object evaluate(final String expression, final Map<String, ?> values) throws ExpressionEvaluationException {
38          LOG.debug("Evaluating Groovy expression: {1}", expression);
39          try {
40              final ObjectCache<String, Script> scriptCache = threadScriptCache.get();
41              Script script = scriptCache.get(expression);
42              if (script == null) {
43                  script = GROOVY_SHELL.parse(expression);
44                  scriptCache.put(expression, script);
45              }
46  
47              final Binding binding = new Binding();
48              for (final Entry<String, ?> entry : values.entrySet()) {
49                  binding.setVariable(entry.getKey(), entry.getValue());
50              }
51              script.setBinding(binding);
52              return script.run();
53          } catch (final Exception ex) {
54              throw new ExpressionEvaluationException("Evaluating script with Groovy failed.", ex);
55          }
56      }
57  }
