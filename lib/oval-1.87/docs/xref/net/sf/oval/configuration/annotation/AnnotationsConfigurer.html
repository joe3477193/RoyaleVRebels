64      private List<ParameterConfiguration> _createParameterConfiguration(final Annotation[][] paramAnnotations, final Class<?>[] parameterTypes) {
65          final CollectionFactory cf = getCollectionFactory();
66  
67          final List<ParameterConfiguration> paramCfg = cf.createList(paramAnnotations.length);
68  
69          List<Check> paramChecks = cf.createList(2);
70          List<CheckExclusion> paramCheckExclusions = cf.createList(2);
71  
72          // loop over all parameters of the current constructor
73          for (int i = 0; i < paramAnnotations.length; i++) {
74              // loop over all annotations of the current constructor parameter
75              for (final Annotation annotation : paramAnnotations[i])
76                  // check if the current annotation is a constraint annotation
77                  if (annotation.annotationType().isAnnotationPresent(Constraint.class)) {
78                      paramChecks.add(initializeCheck(annotation));
79                  } else if (annotation.annotationType().isAnnotationPresent(Constraints.class)) {
80                      initializeChecks(annotation, paramChecks);
81                  } else if (annotation.annotationType().isAnnotationPresent(Exclusion.class)) {
82                      paramCheckExclusions.add(initializeExclusion(annotation));
83                  }
84  
85              final ParameterConfiguration pc = new ParameterConfiguration();
86              paramCfg.add(pc);
87              pc.type = parameterTypes[i];
88              if (paramChecks.size() > 0) {
89                  pc.checks = paramChecks;
90                  paramChecks = cf.createList(2); // create a new list for the next parameter having checks
91              }
92              if (paramCheckExclusions.size() > 0) {
93                  pc.checkExclusions = paramCheckExclusions;
94                  paramCheckExclusions = cf.createList(2); // create a new list for the next parameter having check exclusions
95              }
96          }
97          return paramCfg;
98      }

100     public boolean addCheckInitializationListener(final CheckInitializationListener listener) {
101         Assert.argumentNotNull("listener", "[listener] must not be null");
102         return listeners.add(listener);
103     }

105     protected void configureConstructorParameterChecks(final ClassConfiguration classCfg) {
106         final CollectionFactory cf = getCollectionFactory();
107 
108         for (final Constructor<?> ctor : classCfg.type.getDeclaredConstructors()) {
109             final List<ParameterConfiguration> paramCfg = _createParameterConfiguration(ctor.getParameterAnnotations(), ctor.getParameterTypes());
110 
111             final boolean postValidateThis = ctor.isAnnotationPresent(PostValidateThis.class);
112 
113             if (paramCfg.size() > 0 || postValidateThis) {
114                 if (classCfg.constructorConfigurations == null) {
115                     classCfg.constructorConfigurations = cf.createSet(2);
116                 }
117 
118                 final ConstructorConfiguration cc = new ConstructorConfiguration();
119                 cc.parameterConfigurations = paramCfg;
120                 cc.postCheckInvariants = postValidateThis;
121                 classCfg.constructorConfigurations.add(cc);
122             }
123         }
124     }

126     protected void configureFieldChecks(final ClassConfiguration classCfg) {
127         final CollectionFactory cf = getCollectionFactory();
128 
129         List<Check> checks = cf.createList(2);
130 
131         for (final Field field : classCfg.type.getDeclaredFields()) {
132             // loop over all annotations of the current field
133             for (final Annotation annotation : field.getAnnotations())
134                 // check if the current annotation is a constraint annotation
135                 if (annotation.annotationType().isAnnotationPresent(Constraint.class)) {
136                     checks.add(initializeCheck(annotation));
137                 } else if (annotation.annotationType().isAnnotationPresent(Constraints.class)) {
138                     initializeChecks(annotation, checks);
139                 }
140 
141             if (checks.size() > 0) {
142                 if (classCfg.fieldConfigurations == null) {
143                     classCfg.fieldConfigurations = cf.createSet(2);
144                 }
145 
146                 final FieldConfiguration fc = new FieldConfiguration();
147                 fc.name = field.getName();
148                 fc.checks = checks;
149                 classCfg.fieldConfigurations.add(fc);
150                 checks = cf.createList(2); // create a new list for the next field with checks
151             }
152         }
153     }

155     /**
156      * configure method return value and parameter checks
157      */
158     protected void configureMethodChecks(final ClassConfiguration classCfg) {
159         final CollectionFactory cf = getCollectionFactory();
160 
161         List<Check> returnValueChecks = cf.createList(2);
162         List<PreCheck> preChecks = cf.createList(2);
163         List<PostCheck> postChecks = cf.createList(2);
164 
165         for (final Method method : classCfg.type.getDeclaredMethods()) {
166             /*
167              * determine method return value checks and method pre/post
168              * conditions
169              */
170             boolean preValidateThis = false;
171             boolean postValidateThis = false;
172 
173             // loop over all annotations
174             for (final Annotation annotation : ReflectionUtils.getAnnotations(method, Boolean.TRUE.equals(classCfg.inspectInterfaces)))
175                 if (annotation instanceof Pre) {
176                     final PreCheck pc = new PreCheck();
177                     pc.configure((Pre) annotation);
178                     preChecks.add(pc);
179                 } else if (annotation instanceof PreValidateThis) {
180                     preValidateThis = true;
181                 } else if (annotation instanceof Post) {
182                     final PostCheck pc = new PostCheck();
183                     pc.configure((Post) annotation);
184                     postChecks.add(pc);
185                 } else if (annotation instanceof PostValidateThis) {
186                     postValidateThis = true;
187                 } else if (annotation.annotationType().isAnnotationPresent(Constraint.class)) {
188                     returnValueChecks.add(initializeCheck(annotation));
189                 } else if (annotation.annotationType().isAnnotationPresent(Constraints.class)) {
190                     initializeChecks(annotation, returnValueChecks);
191                 }
192 
193             /*
194              * determine parameter checks
195              */
196             final List<ParameterConfiguration> paramCfg = _createParameterConfiguration(ReflectionUtils.getParameterAnnotations(method, Boolean.TRUE.equals(
197                 classCfg.inspectInterfaces)), method.getParameterTypes());
198 
199             // check if anything has been configured for this method at all
200             if (paramCfg.size() > 0 || returnValueChecks.size() > 0 || preChecks.size() > 0 || postChecks.size() > 0 || preValidateThis || postValidateThis) {
201                 if (classCfg.methodConfigurations == null) {
202                     classCfg.methodConfigurations = cf.createSet(2);
203                 }
204 
205                 final MethodConfiguration mc = new MethodConfiguration();
206                 mc.name = method.getName();
207                 mc.parameterConfigurations = paramCfg;
208                 mc.isInvariant = ReflectionUtils.isAnnotationPresent(method, IsInvariant.class, Boolean.TRUE.equals(classCfg.inspectInterfaces));
209                 mc.preCheckInvariants = preValidateThis;
210                 mc.postCheckInvariants = postValidateThis;
211                 if (returnValueChecks.size() > 0) {
212                     mc.returnValueConfiguration = new MethodReturnValueConfiguration();
213                     mc.returnValueConfiguration.checks = returnValueChecks;
214                     returnValueChecks = cf.createList(2); // create a new list for the next method having return value checks
215                 }
216                 if (preChecks.size() > 0) {
217                     mc.preExecutionConfiguration = new MethodPreExecutionConfiguration();
218                     mc.preExecutionConfiguration.checks = preChecks;
219                     preChecks = cf.createList(2); // create a new list for the next method having pre checks
220                 }
221                 if (postChecks.size() > 0) {
222                     mc.postExecutionConfiguration = new MethodPostExecutionConfiguration();
223                     mc.postExecutionConfiguration.checks = postChecks;
224                     postChecks = cf.createList(2); // create a new list for the next method having post checks
225                 }
226                 classCfg.methodConfigurations.add(mc);
227             }
228         }
229     }

231     protected void configureObjectLevelChecks(final ClassConfiguration classCfg) {
232         final List<Check> checks = getCollectionFactory().createList(2);
233 
234         for (final Annotation annotation : ReflectionUtils.getAnnotations(classCfg.type, Boolean.TRUE.equals(classCfg.inspectInterfaces)))
235             // check if the current annotation is a constraint annotation
236             if (annotation.annotationType().isAnnotationPresent(Constraint.class)) {
237                 checks.add(initializeCheck(annotation));
238             } else if (annotation.annotationType().isAnnotationPresent(Constraints.class)) {
239                 initializeChecks(annotation, checks);
240             }
241 
242         if (checks.size() > 0) {
243             classCfg.objectConfiguration = new ObjectConfiguration();
244             classCfg.objectConfiguration.checks = checks;
245         }
246     }

248     public ClassConfiguration getClassConfiguration(final Class<?> clazz) {
249         final ClassConfiguration classCfg = new ClassConfiguration();
250         classCfg.type = clazz;
251 
252         final Guarded guarded = clazz.getAnnotation(Guarded.class);
253 
254         if (guarded == null) {
255             classCfg.applyFieldConstraintsToConstructors = false;
256             classCfg.applyFieldConstraintsToSetters = false;
257             classCfg.assertParametersNotNull = false;
258             classCfg.checkInvariants = false;
259             classCfg.inspectInterfaces = false;
260         } else {
261             classCfg.applyFieldConstraintsToConstructors = guarded.applyFieldConstraintsToConstructors();
262             classCfg.applyFieldConstraintsToSetters = guarded.applyFieldConstraintsToSetters();
263             classCfg.assertParametersNotNull = guarded.assertParametersNotNull();
264             classCfg.checkInvariants = guarded.checkInvariants();
265             classCfg.inspectInterfaces = guarded.inspectInterfaces();
266         }
267 
268         configureObjectLevelChecks(classCfg);
269         configureFieldChecks(classCfg);
270         configureConstructorParameterChecks(classCfg);
271         configureMethodChecks(classCfg);
272 
273         return classCfg;
274     }

280     protected <ConstraintAnnotation extends Annotation> AnnotationCheck<ConstraintAnnotation> initializeCheck(final ConstraintAnnotation constraintAnnotation)
281             throws ReflectionException {
282         assert constraintAnnotation != null;
283 
284         final Constraint constraint = constraintAnnotation.annotationType().getAnnotation(Constraint.class);
285 
286         // determine the check class
287         @SuppressWarnings("unchecked")
288         final Class<AnnotationCheck<ConstraintAnnotation>> checkClass = (Class<AnnotationCheck<ConstraintAnnotation>>) constraint.checkWith();
289 
290         // instantiate the appropriate check for the found constraint
291         final AnnotationCheck<ConstraintAnnotation> check = newCheckInstance(checkClass);
292         check.configure(constraintAnnotation);
293 
294         for (final CheckInitializationListener listener : listeners) {
295             listener.onCheckInitialized(check);
296         }
297         return check;
298     }

300     /**
301      * handles list of annotations like @Assert.List(...)
302      */
303     protected <ConstraintsAnnotation extends Annotation> void initializeChecks(final ConstraintsAnnotation constraintsAnnotation, final List<Check> checks)
304             throws ReflectionException {
305         try {
306             final Method getValue = constraintsAnnotation.annotationType().getDeclaredMethod("value", (Class<?>[]) null);
307             final Object[] constraintAnnotations = (Object[]) getValue.invoke(constraintsAnnotation, (Object[]) null);
308 
309             final ConstraintsCheck constraintsCheck = new ConstraintsCheck();
310             constraintsCheck.configure(constraintsAnnotation);
311             constraintsCheck.checks = new ArrayList<Check>(constraintAnnotations.length);
312             for (final Object ca : constraintAnnotations) {
313                 constraintsCheck.checks.add(initializeCheck((Annotation) ca));
314             }
315             checks.add(constraintsCheck);
316         } catch (final ReflectionException ex) {
317             throw ex;
318         } catch (final Exception ex) {
319             throw new ReflectionException("Cannot initialize constraint check " + constraintsAnnotation.annotationType().getName(), ex);
320         }
321     }

323     protected <ExclusionAnnotation extends Annotation> AnnotationCheckExclusion<ExclusionAnnotation> initializeExclusion(
324             final ExclusionAnnotation exclusionAnnotation) throws ReflectionException {
325         assert exclusionAnnotation != null;
326 
327         final Exclusion constraint = exclusionAnnotation.annotationType().getAnnotation(Exclusion.class);
328 
329         // determine the check class
330         final Class<?> exclusionClass = constraint.excludeWith();
331 
332         try {
333             // instantiate the appropriate exclusion for the found annotation
334             @SuppressWarnings("unchecked")
335             final AnnotationCheckExclusion<ExclusionAnnotation> exclusion = (AnnotationCheckExclusion<ExclusionAnnotation>) exclusionClass.newInstance();
336             exclusion.configure(exclusionAnnotation);
337             return exclusion;
338         } catch (final Exception ex) {
339             throw new ReflectionException("Cannot initialize constraint exclusion " + exclusionClass.getName(), ex);
340         }
341     }

343     /**
344      * @return a new instance of the given constraint check implementation class
345      */
346     protected <ConstraintAnnotation extends Annotation> AnnotationCheck<ConstraintAnnotation> newCheckInstance(
347             final Class<AnnotationCheck<ConstraintAnnotation>> checkClass) throws OValException {
348         try {
349             return checkClass.newInstance();
350         } catch (final InstantiationException ex) {
351             throw new ReflectionException("Cannot initialize constraint check " + checkClass.getName(), ex);
352         } catch (final IllegalAccessException ex) {
353             throw new ReflectionException("Cannot initialize constraint check " + checkClass.getName(), ex);
354         }
355     }

357     public boolean removeCheckInitializationListener(final CheckInitializationListener listener) {
358         return listeners.remove(listener);
