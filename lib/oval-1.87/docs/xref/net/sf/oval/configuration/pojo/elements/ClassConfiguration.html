14  package net.sf.oval.configuration.pojo.elements;
15  
16  import java.util.Set;
17  
18  /**
19   * @author Sebastian Thomschke
20   */
21  public class ClassConfiguration extends ConfigurationElement {
22      private static final long serialVersionUID = 1L;
23  
24      /**
25       * class type
26       */
27      public Class<?> type;
28  
29      /**
30       * object level constraints configuration
31       */
32      public ObjectConfiguration objectConfiguration;
33  
34      /**
35       * field constraints configuration
36       */
37      public Set<FieldConfiguration> fieldConfigurations;
38  
39      /**
40       * constructor constraints configuration
41       */
42      public Set<ConstructorConfiguration> constructorConfigurations;
43  
44      /**
45       * method constraints configuration
46       */
47      public Set<MethodConfiguration> methodConfigurations;
48  
49      /**
50       * Automatically apply field constraints to the corresponding parameters
51       * of constructors declared within the same class. A corresponding
52       * parameter is a parameter with the same name and type as the field.
53       */
54      public Boolean applyFieldConstraintsToConstructors;
55  
56      /**
57       * Automatically apply field constraints to the parameters of the
58       * corresponding setter methods declared within the same class. A
59       * corresponding setter method is a method following the JavaBean
60       * convention and its parameter has as the same type as the field.
61       */
62      public Boolean applyFieldConstraintsToSetters;
63  
64      /**
65       * Declares if parameter values of constructors and methods are expected to be not null.
66       * This can be weakened by using the @net.sf.oval.constraint.exclusion.Nullable annotation on specific parameters.
67       */
68      public Boolean assertParametersNotNull;
69  
70      /**
71       * Specifies if invariants are checked prior and after calls to
72       * non-private methods and constructors.
73       */
74      public Boolean checkInvariants;
75  
76      /**
77       * Specifies whether annotations can be applied to interfaces that this class implements,
78       * supporting a documentation function
79       */
80      public Boolean inspectInterfaces;
81  }
