13  package net.sf.oval.localization.context;
14  
15  import java.util.Enumeration;
16  import java.util.Locale;
17  import java.util.MissingResourceException;
18  import java.util.ResourceBundle;
19  
20  import net.sf.oval.Validator;
21  import net.sf.oval.context.ClassContext;
22  import net.sf.oval.context.ConstructorParameterContext;
23  import net.sf.oval.context.FieldContext;
24  import net.sf.oval.context.MethodEntryContext;
25  import net.sf.oval.context.MethodExitContext;
26  import net.sf.oval.context.MethodParameterContext;
27  import net.sf.oval.context.MethodReturnValueContext;
28  import net.sf.oval.context.OValContext;
29  import net.sf.oval.internal.Log;
30  
31  /**
32   * This renderer searches for a resource file that is in the same package and has the same name as the validated class.
33   * It then tries to lookup a localized version of the validation context, e.g.<br>
34   * <b>com.acme.model.Person.java<br>
35   * com.acme.model.Person.properties<br>
36   * com.acme.model.Person_de.properties<br>
37   * com.acme.model.Person_fr.properties</b>
38   *
39   * <p>
40   * The properties file is expected to have values following this scheme
41   * 
42   * <pre>
43   * label.class=My translated name of the class name
44   * label.field.firstname=My translated name of the field "firstname"
45   * label.field.lastname=My translated name of the field "lastname"
46   * label.parameter.amount=My translated name of a constructor/method parameter "amount"
47   * label.method.increase=My translated name of the method "increase"
48   * </pre>
49   * 
50   * @author Sebastian Thomschke
51   */
52  public class ResourceBundleValidationContextRenderer implements OValContextRenderer {
53      private static final Log LOG = Log.getLog(ResourceBundleValidationContextRenderer.class);
54  
55      public static final ResourceBundleValidationContextRenderer INSTANCE = new ResourceBundleValidationContextRenderer();
56  
57      private static boolean containsKey(final ResourceBundle bundle, final String key) {
58          for (final Enumeration<String> en = bundle.getKeys(); en.hasMoreElements();)
59              if (en.nextElement().equals(key))
60                  return true;
61          return false;
62      }
63  
64      protected Locale getLocale() {
65          return Validator.getLocaleProvider().getLocale();
66      }
67  
68      public String render(final OValContext ovalContext) {
69          final String baseName;
70          final String key;
71          if (ovalContext instanceof ClassContext) {
72              final ClassContext ctx = (ClassContext) ovalContext;
73              baseName = ctx.getClazz().getName();
74              key = "label.class";
75          } else if (ovalContext instanceof FieldContext) {
76              final FieldContext ctx = (FieldContext) ovalContext;
77              baseName = ctx.getField().getDeclaringClass().getName();
78              final String fieldName = ctx.getField().getName();
79              key = "label.field." + fieldName;
80          } else if (ovalContext instanceof ConstructorParameterContext) {
81              final ConstructorParameterContext ctx = (ConstructorParameterContext) ovalContext;
82              baseName = ctx.getConstructor().getDeclaringClass().getName();
83              key = "label.parameter." + ctx.getParameterName();
84          } else if (ovalContext instanceof MethodParameterContext) {
85              final MethodParameterContext ctx = (MethodParameterContext) ovalContext;
86              baseName = ctx.getMethod().getDeclaringClass().getName();
87              key = "label.parameter." + ctx.getParameterName();
88          } else if (ovalContext instanceof MethodEntryContext) {
89              final MethodEntryContext ctx = (MethodEntryContext) ovalContext;
90              baseName = ctx.getMethod().getDeclaringClass().getName();
91              key = "label.method." + ctx.getMethod().getName();
92          } else if (ovalContext instanceof MethodExitContext) {
93              final MethodExitContext ctx = (MethodExitContext) ovalContext;
94              baseName = ctx.getMethod().getDeclaringClass().getName();
95              key = "label.method." + ctx.getMethod().getName();
96          } else if (ovalContext instanceof MethodReturnValueContext) {
97              final MethodReturnValueContext ctx = (MethodReturnValueContext) ovalContext;
98              baseName = ctx.getMethod().getDeclaringClass().getName();
99              key = "label.method." + ctx.getMethod().getName();
100         } else
101             return ovalContext.toString();
102 
103         try {
104             final ResourceBundle bundle = ResourceBundle.getBundle(baseName, getLocale());
105             if (containsKey(bundle, key))
106                 return bundle.getString(key);
107             LOG.debug("Key {1} not found in bundle {2}", key, baseName);
108         } catch (final MissingResourceException ex) {
109             LOG.debug("Bundle {1} not found", baseName, ex);
110         }
111         return ovalContext.toString();
112     }
113 }
