13  package net.sf.oval.guard;
14  
15  import java.lang.reflect.Constructor;
16  
17  import org.aopalliance.intercept.ConstructorInterceptor;
18  import org.aopalliance.intercept.ConstructorInvocation;
19  import org.aopalliance.intercept.MethodInterceptor;
20  import org.aopalliance.intercept.MethodInvocation;
21  
22  import net.sf.oval.internal.Log;
23  import net.sf.oval.internal.util.Invocable;
24  
25  /**
26   * AOP Alliance Interceptor implementation of the Guard aspect
27   *
28   * @author Sebastian Thomschke
29   */
30  public class GuardInterceptor implements MethodInterceptor, ConstructorInterceptor {
31      protected static final class MethodInvocable implements Invocable {
32          private final MethodInvocation methodInvocation;
33  
34          protected MethodInvocable(final MethodInvocation methodInvocation) {
35              this.methodInvocation = methodInvocation;
36          }
37  
38          public Object invoke() throws Throwable {
39              return methodInvocation.proceed();
40          }
41      }
42  
43      private static final Log LOG = Log.getLog(GuardInterceptor.class);
44  
45      private Guard guard;
46  
47      public GuardInterceptor() {
48          this(new Guard());
49      }
50  
51      public GuardInterceptor(final Guard guard) {
52          LOG.info("Instantiated");
53  
54          setGuard(guard);
55      }
56  
57      public Object construct(final ConstructorInvocation constructorInvocation) throws Throwable {
58          final Constructor<?> ctor = constructorInvocation.getConstructor();
59          final Object[] args = constructorInvocation.getArguments();
60          final Object target = constructorInvocation.getThis();
61  
62          // pre conditions
63          {
64              guard.guardConstructorPre(target, ctor, args);
65          }
66  
67          final Object result = constructorInvocation.proceed();
68  
69          // post conditions
70          {
71              guard.guardConstructorPost(target, ctor, args);
72          }
73  
74          return result;
75      }
76  
77      public Guard getGuard() {
78          return guard;
79      }
80  
81      public Object invoke(final MethodInvocation methodInvocation) throws Throwable {
82          return guard.guardMethod(methodInvocation.getThis(), methodInvocation.getMethod(), methodInvocation.getArguments(), new MethodInvocable(
83              methodInvocation));
84      }
85  
86      public void setGuard(final Guard guard) {
87          this.guard = guard;
88      }
89  }
