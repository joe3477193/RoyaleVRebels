13  package net.sf.oval.guard;
14  
15  import org.aspectj.lang.annotation.Aspect;
16  import org.aspectj.lang.annotation.DeclareWarning;
17  
18  /**
19   * This is an annotations based version of the {@link ApiUsageAuditor} aspect
20   *
21   * @author Sebastian Thomschke
22   */
23  @Aspect
24  public abstract class ApiUsageAuditor2 {
25      /*
26       * Rule 1: Warn about return value constraints for void methods
27       */
28      @DeclareWarning("execution(@(@net.sf.oval.configuration.annotation.Constraint *) void *.*(..))")
29      public static final String RULE1 = "OVal API usage violation 1: Method return value constraints are not allowed for methods without return values";
30  
31      /*
32       * Rule 2: Warn about return value constraints for non-void, parameterized methods in classes that are not guarded
33       */
34      @DeclareWarning("execution(@(@net.sf.oval.configuration.annotation.Constraint *) !void (!@net.sf.oval.guard.Guarded *).*(*,..))")
35      public static final String RULE2 = "OVal API usage violation 2: Method return value constraints for parameterized methods are only allowed in guarded classes";
36  
37      /*
38       * Rule 3: Warn about return value constraints for non-void, non-parameterized methods missing the @Invariant annotation in classes
39       * that are not guarded
40       */
41      @DeclareWarning("execution(!@net.sf.oval.configuration.annotation.IsInvariant @(@net.sf.oval.configuration.annotation.Constraint *) !void (!@net.sf.oval.guard.Guarded *).*())")
42      public static final String RULE3 = "OVal API usage violation 3: Method return value constraints are only allowed if the method is annotated with @IsInvariant or the declaring class is guarded";
43  
44      /*
45       * Rule 4: Warn about the @PreValidateThis annotation used on methods in classes that are not guarded
46       */
47      @DeclareWarning("execution (@net.sf.oval.guard.PreValidateThis * (!@net.sf.oval.guard.Guarded *).*(..))")
48      public static final String RULE4 = "OVal API usage violation 4: @PreValidateThis is only allowed in guarded classes";
49  
50      /*
51       * Rule 5: Warn about the @PostValidateObject annotation used on methods and constructors in classes not annotated with @Guarded
52       */
53      @DeclareWarning("execution (@net.sf.oval.guard.PostValidateThis * (!@net.sf.oval.guard.Guarded *).*(..)) || execution (@net.sf.oval.guard.PostValidateObject (!@net.sf.oval.guard.Guarded *).new(..))")
54      public static final String RULE5 = "OVal API usage violation 5: @PostValidateThis is only allowed in guarded classes";
55  
56      /*
57       * Rule 6: Warn about method parameter constraints in classes that are not guarded
58       */
59      @DeclareWarning("execution(* (!@net.sf.oval.guard.Guarded *).*(@(@net.sf.oval.configuration.annotation.Constraint *) *, ..)) || execution(* (!@net.sf.oval.guard.Guarded *).*(*, @(@net.sf.oval.configuration.annotation.Constraint *) *, ..)) || execution(* (!@net.sf.oval.guard.Guarded *).*(*, *, @(@net.sf.oval.configuration.annotation.Constraint *) *, ..)) || execution(* (!@net.sf.oval.guard.Guarded *).*(*, *, *, @(@net.sf.oval.configuration.annotation.Constraint *) *, ..)) || execution(* (!@net.sf.oval.guard.Guarded *).*(*, *, *, *, @(@net.sf.oval.configuration.annotation.Constraint *) *, ..)) || execution(* (!@net.sf.oval.guard.Guarded *).*(*, *, *, *, *, @(@net.sf.oval.configuration.annotation.Constraint *) *, ..))")
60      public static final String RULE6 = "OVal API usage violation 6: Method parameter constraints are only allowed in guarded classes";
61  
62      /*
63       * Rule 7: Warn about constructor parameter constraints in classes that are not guarded
64       */
65      @DeclareWarning("execution((!@net.sf.oval.guard.Guarded *).new(@(@net.sf.oval.configuration.annotation.Constraint *) *, ..)) || execution((!@net.sf.oval.guard.Guarded *).new(*, @(@net.sf.oval.configuration.annotation.Constraint *) *, ..)) || execution((!@net.sf.oval.guard.Guarded *).new(*, *, @(@net.sf.oval.configuration.annotation.Constraint *) *, ..)) || execution((!@net.sf.oval.guard.Guarded *).new(*, *, *, @(@net.sf.oval.configuration.annotation.Constraint *) *, ..)) || execution((!@net.sf.oval.guard.Guarded *).new(*, *, *, *, @(@net.sf.oval.configuration.annotation.Constraint *) *, ..)) || execution((!@net.sf.oval.guard.Guarded *).new(*, *, *, *, *, @(@net.sf.oval.configuration.annotation.Constraint *) *, ..))")
66      public static final String RULE7 = "OVal API usage violation 7: Method parameter constraints are only allowed in guarded classes";
67  
68  }
