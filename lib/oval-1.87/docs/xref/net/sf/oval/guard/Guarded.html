14  package net.sf.oval.guard;
15  
16  import java.lang.annotation.Documented;
17  import java.lang.annotation.ElementType;
18  import java.lang.annotation.Retention;
19  import java.lang.annotation.RetentionPolicy;
20  import java.lang.annotation.Target;
21  
22  /**
23   * This annotation needs to be applied to classes where
24   * OVal's programming by contract features shall be used.<br>
25   * <br>
26   * The GuardAspect will weave the required AOP code into all
27   * classes annotated with @Guarded.
28   * 
29   * @author Sebastian Thomschke
30   */
31  @Documented
32  @Retention(RetentionPolicy.RUNTIME)
33  @Target(ElementType.TYPE)
34  public @interface Guarded {
35      /**
36       * Automatically apply field constraints to the corresponding parameters of
37       * constructors declared within the same class.
38       * A corresponding parameter is a parameter with the same name and type as the field.
39       */
40      boolean applyFieldConstraintsToConstructors() default false;
41  
42      /**
43       * Automatically apply field constraints to the single parameter of the corresponding
44       * setter methods declared within the same class.
45       * A corresponding setter method is a method following the JavaBean convention and
46       * its parameter has as the same type as the field.
47       */
48      boolean applyFieldConstraintsToSetters() default false;
49  
50      /**
51       * Declares if parameter values of constructors and methods are expected to be not null.
52       * This can be weakened by using the @net.sf.oval.constraint.exclusion.Nullable annotation on specific parameters.
53       */
54      boolean assertParametersNotNull() default false;
55  
56      /**
57       * Declares if invariants are automatically checked after constructor execution and
58       * prior and after calls to non-private methods.
59       */
60      boolean checkInvariants() default true;
61  
62      /**
63       * Declares if annotations can be applied to interfaces that this class implements - supporting a documentation
64       * function
65       */
66      boolean inspectInterfaces() default false;
67  }
