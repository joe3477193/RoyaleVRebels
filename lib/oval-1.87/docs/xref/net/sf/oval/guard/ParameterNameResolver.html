13  package net.sf.oval.guard;
14  
15  import java.lang.reflect.Constructor;
16  import java.lang.reflect.Method;
17  
18  import net.sf.oval.exception.ReflectionException;
19  
20  /**
21   * @author Sebastian Thomschke
22   */
23  public interface ParameterNameResolver {
24      /**
25       * Returns an empty String array if no parameters are declared.
26       * 
27       * @param constructor the constructor
28       * @return an array holding the parameter names of the given constructor.
29       * @throws ReflectionException in case retrieving the parameter names fails
30       */
31      String[] getParameterNames(Constructor<?> constructor) throws ReflectionException;
32  
33      /**
34       * Returns an empty String array if no parameters are declared
35       * 
36       * @param method the method
37       * @return an array holding the parameter names of the given method.
38       * @throws ReflectionException in case retrieving the parameter names fails
39       */
40      String[] getParameterNames(Method method) throws ReflectionException;
41  }
