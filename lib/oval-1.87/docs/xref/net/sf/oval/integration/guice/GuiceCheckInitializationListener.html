13  package net.sf.oval.integration.guice;
14  
15  import net.sf.oval.Check;
16  import net.sf.oval.configuration.CheckInitializationListener;
17  import net.sf.oval.constraint.CheckWithCheck;
18  
19  import com.google.inject.Injector;
20  
21  /**
22   * Injects Guice bean dependencies into {@link Check} instances.
23   *
24   * Required dependencies must be annotated with <code>@Autowired</code> within {@link Check} the class.
25   *
26   * @author Sebastian Thomschke
27   */
28  public class GuiceCheckInitializationListener implements CheckInitializationListener {
29      private final Injector injector;
30  
31      public GuiceCheckInitializationListener(final Injector injector) {
32          this.injector = injector;
33      }
34  
35      public void onCheckInitialized(final Check check) {
36          injector.injectMembers(check);
37  
38          if (check instanceof CheckWithCheck) {
39              final CheckWithCheck checkWith = (CheckWithCheck) check;
40              injector.injectMembers(checkWith.getSimpleCheck());
41          }
42      }
43  }
