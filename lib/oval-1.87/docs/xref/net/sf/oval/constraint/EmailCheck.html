13  package net.sf.oval.constraint;
14  
15  import java.util.regex.Pattern;
16  
17  import net.sf.oval.ConstraintTarget;
18  import net.sf.oval.Validator;
19  import net.sf.oval.configuration.annotation.AbstractAnnotationCheck;
20  import net.sf.oval.context.OValContext;
21  import net.sf.oval.exception.OValException;
22  
23  /**
24   * @author Sebastian Thomschke
25   */
26  public class EmailCheck extends AbstractAnnotationCheck<Email> {
27      private static final long serialVersionUID = 1L;
28  
29      private static final String SPECIAL_CHARACTERS = "'\\(\\)\\-\\.`";
30      private static final String ASCII = "\\w " + SPECIAL_CHARACTERS;
31      private static final String ASCII_WITHOUT_COMMA = "[" + ASCII + "]+";
32      private static final String ASCII_WITH_COMMA = "\"[" + ASCII + ",]+\"";
33      private static final String ASCII_WITH_QUESTION_MARK_AND_EQUALS = "[" + ASCII + "\\?\\=]+";
34      private static final String MIME_ENCODED = "\\=\\?" + ASCII_WITH_QUESTION_MARK_AND_EQUALS + "\\?\\=";
35      private static final String NAME = "(" + ASCII_WITHOUT_COMMA + "|" + ASCII_WITH_COMMA + "|" + MIME_ENCODED + ")";
36  
37      private static final String EMAIL_BASE_PATTERN = "['_A-Za-z0-9-&+]+(\\.['_A-Za-z0-9-&+]+)*[.]{0,1}@([A-Za-z0-9-])+(\\.[A-Za-z0-9-]+)*((\\.[A-Za-z0-9]{2,})|(\\.[A-Za-z0-9]{2,}\\.[A-Za-z0-9]{2,}))";
38  
39      private static final Pattern EMAIL_PATTERN = Pattern.compile("^" + EMAIL_BASE_PATTERN + "$");
40  
41      private static final Pattern EMAIL_WITH_PERSONAL_NAME_PATTERN = Pattern.compile("^(" + EMAIL_BASE_PATTERN + "|" + NAME + " +<" + EMAIL_BASE_PATTERN
42              + ">)$");
43  
44      private boolean allowPersonalName;
45  
46      @Override
47      public void configure(final Email constraintAnnotation) {
48          super.configure(constraintAnnotation);
49          setAllowPersonalName(constraintAnnotation.allowPersonalName());
50      }
51  
52      @Override
53      protected ConstraintTarget[] getAppliesToDefault() {
54          return new ConstraintTarget[] { ConstraintTarget.VALUES };
55      }
56  
57      public boolean isAllowPersonalName() {
58          return allowPersonalName;
59      }
60  
61      public boolean isSatisfied(final Object validatedObject, final Object valueToValidate, final OValContext context, final Validator validator)
62              throws OValException {
63          if (valueToValidate == null)
64              return true;
65  
66          if (allowPersonalName)
67              return EMAIL_WITH_PERSONAL_NAME_PATTERN.matcher(valueToValidate.toString()).matches();
68          return EMAIL_PATTERN.matcher(valueToValidate.toString()).matches();
69      }
70  
71      public void setAllowPersonalName(final boolean allowPersonalName) {
72          this.allowPersonalName = allowPersonalName;
73      }
74  }
