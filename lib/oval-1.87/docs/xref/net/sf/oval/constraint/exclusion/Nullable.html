13  package net.sf.oval.constraint.exclusion;
14  
15  import java.lang.annotation.Documented;
16  import java.lang.annotation.ElementType;
17  import java.lang.annotation.Retention;
18  import java.lang.annotation.RetentionPolicy;
19  import java.lang.annotation.Target;
20  
21  import net.sf.oval.configuration.annotation.Exclusion;
22  
23  /**
24   * Allows a value to be nullable.
25   * 
26   * @author Sebastian Thomschke
27   */
28  @Documented
29  @Retention(RetentionPolicy.RUNTIME)
30  @Target({ ElementType.PARAMETER })
31  @Exclusion(excludeWith = NullableExclusion.class)
32  public @interface Nullable {
33      /**
34       * The associated constraint profiles.
35       */
36      String[] profiles() default {};
37  
38      /**
39       * Formula returning <code>true</code> if this constraint exclusion shall be active and
40       * <code>false</code> if it shall be ignored for the current validation.
41       * <p>
42       * <b>Important:</b> The formula must be prefixed with the name of the scripting language that is used.
43       * E.g. <code>groovy:_this.amount > 10</code>
44       * <p>
45       * Available context variables are:<br>
46       * <b>_this</b> -&gt; the validated bean<br>
47       * <b>_value</b> -&gt; the value to validate (e.g. the field value, parameter value, method return value,
48       * or the validated bean for object level constraints)
49       */
50      String when() default "";
51  }
