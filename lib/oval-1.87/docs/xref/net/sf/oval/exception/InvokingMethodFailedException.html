13  package net.sf.oval.exception;
14  
15  import net.sf.oval.context.OValContext;
16  import net.sf.oval.internal.MessageRenderer;
17  
18  /**
19   * @author Sebastian Thomschke
20   */
21  public class InvokingMethodFailedException extends ReflectionException {
22      private static final long serialVersionUID = 1L;
23  
24      private final OValContext context;
25      private final Object validatedObject;
26  
27      public InvokingMethodFailedException(final String methodName, final Object validatedObject, final OValContext context, final Throwable cause) {
28          super(MessageRenderer.renderMessage("net.sf.oval.exception.InvokingMethodFailedException.message", "methodName", methodName), cause);
29          this.context = context;
30          this.validatedObject = validatedObject;
31      }
32  
33      public OValContext getContext() {
34          return context;
35      }
36  
37      public Object getValidatedObject() {
38          return validatedObject;
39      }
40  }
