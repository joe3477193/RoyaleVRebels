13  package net.sf.oval.exception;
14  
15  import java.util.List;
16  
17  import net.sf.oval.ConstraintViolation;
18  
19  /**
20   * This exception is thrown if one or more constraints are not satisfied during validation.
21   *
22   * @author Sebastian Thomschke
23   */
24  public class ConstraintsViolatedException extends OValException {
25      private static final long serialVersionUID = 1L;
26  
27      private final long causingThreadId = Thread.currentThread().getId();
28  
29      private final ConstraintViolation[] constraintViolations;
30  
31      /**
32       * @param constraintViolations must not be null
33       */
34      public ConstraintsViolatedException(final ConstraintViolation... constraintViolations) {
35          // the message of the first occurring constraint violation will be used
36          super(constraintViolations[0].getMessage());
37  
38          this.constraintViolations = constraintViolations;
39      }
40  
41      /**
42       * @param constraintViolations must not be null
43       */
44      public ConstraintsViolatedException(final List<ConstraintViolation> constraintViolations) {
45          // the message of the first occurring constraint violation will be used
46          super(constraintViolations.get(0).getMessage());
47  
48          this.constraintViolations = constraintViolations.toArray(new ConstraintViolation[constraintViolations.size()]);
49      }
50  
51      /**
52       * @return the id of the thread in which the violations occurred
53       */
54      public long getCausingThreadId() {
55          return causingThreadId;
56      }
57  
58      public ConstraintViolation[] getConstraintViolations() {
59          return constraintViolations.clone();
60      }
61  }
