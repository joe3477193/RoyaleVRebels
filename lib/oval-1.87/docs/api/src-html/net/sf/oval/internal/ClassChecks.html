124        private void _addConstructorParameterCheckExclusions(final Constructor<?> constructor, final int parameterIndex, final Object exclusions)
125                throws InvalidConfigurationException {
126            final ParameterChecks checksOfConstructorParameter = _getChecksOfConstructorParameter(constructor, parameterIndex);
127    
128            if (exclusions instanceof Collection<?>) {
129                @SuppressWarnings("unchecked")
130                final Collection<CheckExclusion> exclusionsColl = (Collection<CheckExclusion>) exclusions;
131                checksOfConstructorParameter.checkExclusions.addAll(exclusionsColl);
132            } else {
133                ArrayUtils.addAll(checksOfConstructorParameter.checkExclusions, (CheckExclusion[]) exclusions);
134            }
135        }

137        @SuppressWarnings("unchecked")
138        private void _addConstructorParameterChecks(final Constructor<?> constructor, final int parameterIndex, final Object checks)
139                throws InvalidConfigurationException {
140            if (LOG.isDebug() && !IsGuarded.class.isAssignableFrom(clazz)) {
141                LOG.warn("Constructor parameter constraints may not be validated." + GUARDING_MAY_NOT_BE_ACTIVATED_MESSAGE);
142            }
143    
144            final ParameterChecks checksOfConstructorParameter = _getChecksOfConstructorParameter(constructor, parameterIndex);
145    
146            if (checks instanceof Collection) {
147                for (final Check check : (Collection<Check>) checks) {
148                    checksOfConstructorParameter.checks.add(check);
149                    if (check.getContext() == null) {
150                        check.setContext(checksOfConstructorParameter.context);
151                    }
152                }
153            } else {
154                for (final Check check : (Check[]) checks) {
155                    checksOfConstructorParameter.checks.add(check);
156                    if (check.getContext() == null) {
157                        check.setContext(checksOfConstructorParameter.context);
158                    }
159                }
160            }
161        }

163        @SuppressWarnings("unchecked")
164        private void _addFieldChecks(final Field field, final Object checks) {
165            synchronized (checksForFields) {
166                Set<Check> checksOfField = checksForFields.get(field);
167                if (checksOfField == null) {
168                    checksOfField = new LinkedHashSet<Check>(2);
169                    checksForFields.put(field, checksOfField);
170                    if (ReflectionUtils.isStatic(field)) {
171                        constrainedStaticFields.add(field);
172                    } else {
173                        constrainedFields.add(field);
174                    }
175                }
176    
177                if (checks instanceof Collection) {
178                    for (final Check check : (Collection<Check>) checks) {
179                        checksOfField.add(check);
180                        if (check.getContext() == null) {
181                            check.setContext(ContextCache.getFieldContext(field));
182                        }
183                    }
184                } else {
185                    for (final Check check : (Check[]) checks) {
186                        checksOfField.add(check);
187                        if (check.getContext() == null) {
188                            check.setContext(ContextCache.getFieldContext(field));
189                        }
190                    }
191                }
192            }
193        }

295        @SuppressWarnings("unchecked")
296        private void _addMethodReturnValueChecks(final Method method, final Boolean isInvariant, final Object checks) throws InvalidConfigurationException {
297            // ensure the method has a return type
298            if (method.getReturnType() == Void.TYPE)
299                throw new InvalidConfigurationException("Adding return value constraints for method " + method
300                        + " is not possible. The method is declared as void and does not return any values.");
301    
302            if (ReflectionUtils.isVoidMethod(method))
303                throw new InvalidConfigurationException("Cannot apply method return value constraints for void method " + method);
304    
305            final boolean hasParameters = method.getParameterTypes().length > 0;
306    
307            if (LOG.isDebug() && hasParameters && !IsGuarded.class.isAssignableFrom(clazz)) {
308                LOG.warn("Method return value constraints may not be validated." + GUARDING_MAY_NOT_BE_ACTIVATED_MESSAGE);
309            }
310    
311            final boolean isInvariant2 = isInvariant == null ? constrainedMethods.contains(method) : isInvariant;
312    
313            if (LOG.isDebug() && !isInvariant2 && !IsGuarded.class.isAssignableFrom(clazz)) {
314                LOG.warn("Method return value constraints may not be validated." + GUARDING_MAY_NOT_BE_ACTIVATED_MESSAGE);
315            }
316    
317            synchronized (checksForMethodReturnValues) {
318                if (!hasParameters && isInvariant2) {
319                    if (ReflectionUtils.isStatic(method)) {
320                        constrainedStaticMethods.add(method);
321                    } else {
322                        constrainedMethods.add(method);
323                    }
324                } else if (ReflectionUtils.isStatic(method)) {
325                    constrainedStaticMethods.remove(method);
326                } else {
327                    constrainedMethods.remove(method);
328                }
329    
330                Set<Check> methodChecks = checksForMethodReturnValues.get(method);
331                if (methodChecks == null) {
332                    methodChecks = new LinkedHashSet<Check>(2);
333                    checksForMethodReturnValues.put(method, methodChecks);
334                }
335    
336                if (checks instanceof Collection) {
337                    for (final Check check : (Collection<Check>) checks) {
338                        methodChecks.add(check);
339                        if (check.getContext() == null) {
340                            check.setContext(ContextCache.getMethodReturnValueContext(method));
341                        }
342                    }
343                } else {
344                    for (final Check check : (Check[]) checks) {
345                        methodChecks.add(check);
346                        if (check.getContext() == null) {
347                            check.setContext(ContextCache.getMethodReturnValueContext(method));
348                        }
349                    }
350                }
351            }
352        }

354        private ParameterChecks _getChecksOfConstructorParameter(final Constructor<?> ctor, final int paramIndex) {
355            final int paramCount = ctor.getParameterTypes().length;
356    
357            if (paramIndex < 0 || paramIndex >= paramCount)
358                throw new InvalidConfigurationException("Parameter Index " + paramIndex + " is out of range (0-" + (paramCount - 1) + ")");
359    
360            synchronized (checksForConstructorParameters) {
361                // retrieve the currently registered checks for all parameters of the specified constructor
362                Map<Integer, ParameterChecks> checksOfConstructorByParameter = checksForConstructorParameters.get(ctor);
363                if (checksOfConstructorByParameter == null) {
364                    checksOfConstructorByParameter = getCollectionFactory().createMap(paramCount);
365                    checksForConstructorParameters.put(ctor, checksOfConstructorByParameter);
366                }
367    
368                // retrieve the checks for the specified parameter
369                ParameterChecks checksOfConstructorParameter = checksOfConstructorByParameter.get(paramIndex);
370                if (checksOfConstructorParameter == null) {
371                    checksOfConstructorParameter = new ParameterChecks(ctor, paramIndex, parameterNameResolver.getParameterNames(ctor)[paramIndex]);
372                    checksOfConstructorByParameter.put(paramIndex, checksOfConstructorParameter);
373                }
374    
375                return checksOfConstructorParameter;
376            }
377        }

379        private ParameterChecks _getChecksOfMethodParameter(final Method method, final int paramIndex) {
380            final int paramCount = method.getParameterTypes().length;
381    
382            if (paramIndex < 0 || paramIndex >= paramCount)
383                throw new InvalidConfigurationException("Parameter index " + paramIndex + " is out of range (0-" + (paramCount - 1) + ")");
384    
385            synchronized (checksForMethodParameters) {
386                // retrieve the currently registered checks for all parameters of the specified method
387                Map<Integer, ParameterChecks> checksOfMethodByParameter = checksForMethodParameters.get(method);
388                if (checksOfMethodByParameter == null) {
389                    checksOfMethodByParameter = getCollectionFactory().createMap(paramCount);
390                    checksForMethodParameters.put(method, checksOfMethodByParameter);
391                }
392    
393                // retrieve the checks for the specified parameter
394                ParameterChecks checksOfMethodParameter = checksOfMethodByParameter.get(paramIndex);
395                if (checksOfMethodParameter == null) {
396                    checksOfMethodParameter = new ParameterChecks(method, paramIndex, parameterNameResolver.getParameterNames(method)[paramIndex]);
397                    checksOfMethodByParameter.put(paramIndex, checksOfMethodParameter);
398                }
399    
400                return checksOfMethodParameter;
401            }

579            checksForMethodsPostExcecution.clear();
580            checksForMethodsPreExecution.clear();

601        public void clearConstructorParameterChecks(final Constructor<?> constructor, final int parameterIndex) {
602            synchronized (checksForConstructorParameters) {
603                // retrieve the currently registered checks for all parameters of the specified method
604                final Map<Integer, ParameterChecks> checksOfConstructorByParameter = checksForConstructorParameters.get(constructor);
605                if (checksOfConstructorByParameter == null)
606                    return;
607    
608                // retrieve the checks for the specified parameter
609                final ParameterChecks checksOfMethodParameter = checksOfConstructorByParameter.get(parameterIndex);
610                if (checksOfMethodParameter == null)
611                    return;
612    
613                checksOfConstructorByParameter.remove(parameterIndex);
614            }
615        }
