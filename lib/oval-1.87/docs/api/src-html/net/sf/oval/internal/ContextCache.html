013    package net.sf.oval.internal;
014    
015    import java.lang.reflect.Field;
016    import java.lang.reflect.Method;
017    import java.util.Map;
018    import java.util.WeakHashMap;
019    
020    import net.sf.oval.context.ClassContext;
021    import net.sf.oval.context.FieldContext;
022    import net.sf.oval.context.MethodEntryContext;
023    import net.sf.oval.context.MethodExitContext;
024    import net.sf.oval.context.MethodReturnValueContext;
025    
026    /**
027     * @author Sebastian Thomschke
028     */
029    public final class ContextCache {
030        private static final Map<Class<?>, ClassContext> CLASS_CONTEXTS = new WeakHashMap<Class<?>, ClassContext>();
031        private static final Map<Field, FieldContext> FIELD_CONTEXTS = new WeakHashMap<Field, FieldContext>();
032        private static final Map<Method, MethodEntryContext> METHOD_ENTRY_CONTEXTS = new WeakHashMap<Method, MethodEntryContext>();
033        private static final Map<Method, MethodExitContext> METHOD_EXIT_CONTEXTS = new WeakHashMap<Method, MethodExitContext>();
034        private static final Map<Method, MethodReturnValueContext> METHOD_RETURN_VALUE_CONTEXTS = new WeakHashMap<Method, MethodReturnValueContext>();
035    
036        public static ClassContext getClassContext(final Class<?> clazz) {
037            synchronized (CLASS_CONTEXTS) {
038                ClassContext ctx = CLASS_CONTEXTS.get(clazz);
039                if (ctx == null) {
040                    ctx = new ClassContext(clazz);
041                    CLASS_CONTEXTS.put(clazz, ctx);
042                }
043                return ctx;
044            }
045        }
046    
047        public static FieldContext getFieldContext(final Field field) {
048            synchronized (FIELD_CONTEXTS) {
049                FieldContext ctx = FIELD_CONTEXTS.get(field);
050                if (ctx == null) {
051                    ctx = new FieldContext(field);
052                    FIELD_CONTEXTS.put(field, ctx);
053                }
054                return ctx;
055            }
056        }
057    
058        public static MethodEntryContext getMethodEntryContext(final Method method) {
059            synchronized (METHOD_ENTRY_CONTEXTS) {
060                MethodEntryContext ctx = METHOD_ENTRY_CONTEXTS.get(method);
061                if (ctx == null) {
062                    ctx = new MethodEntryContext(method);
063                    METHOD_ENTRY_CONTEXTS.put(method, ctx);
064                }
065                return ctx;
066            }
067        }
068    
069        public static MethodExitContext getMethodExitContext(final Method method) {
070            synchronized (METHOD_EXIT_CONTEXTS) {
071                MethodExitContext ctx = METHOD_EXIT_CONTEXTS.get(method);
072                if (ctx == null) {
073                    ctx = new MethodExitContext(method);
074                    METHOD_EXIT_CONTEXTS.put(method, ctx);
075                }
076                return ctx;
077            }
078        }
079    
080        public static MethodReturnValueContext getMethodReturnValueContext(final Method method) {
081            synchronized (METHOD_RETURN_VALUE_CONTEXTS) {
082                MethodReturnValueContext ctx = METHOD_RETURN_VALUE_CONTEXTS.get(method);
083                if (ctx == null) {
084                    ctx = new MethodReturnValueContext(method);
085                    METHOD_RETURN_VALUE_CONTEXTS.put(method, ctx);
086                }
087                return ctx;
088            }
089        }
090    
091        private ContextCache() {
092            super();
093        }
094    }
