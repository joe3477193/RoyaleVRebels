013    package net.sf.oval.configuration.annotation;
014    
015    import java.lang.annotation.Annotation;
016    import java.lang.reflect.Method;
017    
018    import net.sf.oval.AbstractCheck;
019    import net.sf.oval.ConstraintTarget;
020    import net.sf.oval.internal.Log;
021    import net.sf.oval.internal.util.ReflectionUtils;
022    
023    /**
024     * Partial implementation of check classes configurable via annotations.
025     *
026     * @author Sebastian Thomschke
027     */
028    public abstract class AbstractAnnotationCheck<ConstraintAnnotation extends Annotation> extends AbstractCheck implements AnnotationCheck<ConstraintAnnotation> {
029        private static final long serialVersionUID = 1L;
030    
031        private static final Log LOG = Log.getLog(AbstractAnnotationCheck.class);
032    
033        public void configure(final ConstraintAnnotation constraintAnnotation) {
034            final Class<?> constraintClazz = constraintAnnotation.getClass();
035    
036            /*
037             * Retrieve the message value from the constraint annotation via reflection.
038             * Using reflection is required because annotations do not support inheritance and
039             * therefore cannot implement an interface that could be used for a down cast here.
040             */
041            final Method getMessage = ReflectionUtils.getMethod(constraintClazz, "message", (Class<?>[]) null);
042            if (getMessage == null) {
043                LOG.debug("Cannot determine constraint error message based on annotation {1} since attribtue message() is not defined.", constraintClazz.getName());
044            } else {
045                try {
046                    setMessage((String) getMessage.invoke(constraintAnnotation, (Object[]) null));
047                } catch (final Exception ex) {
048                    LOG.warn("Cannot determine constraint error message based on annotation {1}", constraintClazz.getName(), ex);
049    
050                    try {
051                        setMessage(constraintClazz.getName() + ".violated");
052                    } catch (final UnsupportedOperationException uex) {
053                        // ignore
054                    }
055                }
056            }
057    
058            /*
059             * Retrieve the appliesTo value from the constraint annotation via reflection.
060             */
061            final Method getAppliesTo = ReflectionUtils.getMethod(constraintClazz, "appliesTo", (Class<?>[]) null);
062            if (getAppliesTo == null) {
063                LOG.debug("Cannot determine constraint targets based on annotation {1} since attribtue appliesTo() is not defined.", constraintClazz.getName());
064            } else {
065                try {
066                    setAppliesTo((ConstraintTarget[]) getAppliesTo.invoke(constraintAnnotation, (Object[]) null));
067                } catch (final Exception ex) {
068                    LOG.warn("Cannot determine constraint targets based on annotation {1}", constraintClazz.getName(), ex);
069                }
070            }
071    
072            /*
073             * Retrieve the error code value from the constraint annotation via reflection.
074             */
075            final Method getErrorCode = ReflectionUtils.getMethod(constraintClazz, "errorCode", (Class<?>[]) null);
076            if (getErrorCode == null) {
077                LOG.debug("Cannot determine constraint error code based on annotation {1} since attribtue errorCode() is not defined.", constraintClazz.getName());
078            } else {
079                try {
080                    setErrorCode((String) getErrorCode.invoke(constraintAnnotation, (Object[]) null));
081                } catch (final Exception ex) {
082                    LOG.warn("Cannot determine constraint error code based on annotation {1}", constraintClazz.getName(), ex);
083                    try {
084                        setErrorCode(constraintClazz.getName());
085                    } catch (final UnsupportedOperationException uex) {
086                        // ignore
087                    }
088                }
089            }
090    
091            /*
092             * Retrieve the severity value from the constraint annotation via reflection.
093             */
094            final Method getSeverity = ReflectionUtils.getMethod(constraintClazz, "severity", (Class<?>[]) null);
095            if (getSeverity == null) {
096                LOG.debug("Cannot determine constraint severity based on annotation {1} since attribtue severity() is not defined.", constraintClazz.getName());
097            } else {
098                try {
099                    setSeverity(((Number) getSeverity.invoke(constraintAnnotation, (Object[]) null)).intValue());
100                } catch (final Exception ex) {
101                    LOG.warn("Cannot determine constraint severity based on annotation {1}", constraintClazz.getName(), ex);
102                }
103            }
104    
105            /*
106             * Retrieve the profiles value from the constraint annotation via reflection.
107             */
108            final Method getProfiles = ReflectionUtils.getMethod(constraintClazz, "profiles", (Class<?>[]) null);
109            if (getProfiles == null) {
110                LOG.debug("Cannot determine constraint profiles based on annotation {1} since attribtue profiles() is not defined.", constraintClazz.getName());
111            } else {
112                try {
113                    setProfiles((String[]) getProfiles.invoke(constraintAnnotation, (Object[]) null));
114                } catch (final Exception ex) {
115                    LOG.warn("Cannot determine constraint profiles based on annotation {1}", constraintClazz.getName(), ex);
116                }
117            }
118    
119            /*
120             * Retrieve the profiles value from the constraint annotation via reflection.
121             */
122            final Method getTarget = ReflectionUtils.getMethod(constraintClazz, "target", (Class<?>[]) null);
123            if (getTarget == null) {
124                LOG.debug("Cannot determine constraint target based on annotation {1} since attribtue target() is not defined.", constraintClazz.getName());
125            } else {
126                try {
127                    setTarget((String) getTarget.invoke(constraintAnnotation, (Object[]) null));
128                } catch (final Exception ex) {
129                    LOG.warn("Cannot determine constraint target based on annotation {1}", constraintClazz.getName(), ex);
130                }
131            }
132    
133            /*
134             * Retrieve the when formula from the constraint annotation via reflection.
135             */
136            final Method getWhen = ReflectionUtils.getMethod(constraintClazz, "when", (Class<?>[]) null);
137            if (getWhen == null) {
138                LOG.debug("Cannot determine constraint when formula based on annotation {1} since attribtue when() is not defined.", constraintClazz.getName());
139            } else {
140                try {
141                    setWhen((String) getWhen.invoke(constraintAnnotation, (Object[]) null));
142                } catch (final Exception ex) {
143                    LOG.warn("Cannot determine constraint when formula based on annotation {1}", constraintClazz.getName(), ex);
144                }
145            }
146        }
147    }
