013    package net.sf.oval.expression;
014    
015    import java.util.ArrayList;
016    import java.util.Map;
017    import java.util.Map.Entry;
018    
019    import net.sf.oval.exception.ExpressionEvaluationException;
020    import net.sf.oval.internal.Log;
021    
022    import org.jruby.CompatVersion;
023    import org.jruby.Ruby;
024    import org.jruby.RubyInstanceConfig;
025    import org.jruby.javasupport.JavaEmbedUtils;
026    import org.jruby.runtime.builtin.IRubyObject;
027    
028    /**
029     * @author Sebastian Thomschke
030     *
031     */
032    public class ExpressionLanguageJRubyImpl extends AbstractExpressionLanguage {
033        private static final Log LOG = Log.getLog(ExpressionLanguageJRubyImpl.class);
034    
035        public Object evaluate(final String expression, final Map<String, ?> values) throws ExpressionEvaluationException {
036            LOG.debug("Evaluating JRuby expression: {1}", expression);
037            try {
038                final RubyInstanceConfig config = new RubyInstanceConfig();
039                config.setCompatVersion(CompatVersion.RUBY1_9);
040                final Ruby runtime = JavaEmbedUtils.initialize(new ArrayList<String>(), config);
041    
042                final StringBuilder localVars = new StringBuilder();
043                for (final Entry<String, ?> entry : values.entrySet()) {
044                    runtime.getGlobalVariables().set("$" + entry.getKey(), JavaEmbedUtils.javaToRuby(runtime, entry.getValue()));
045                    localVars.append(entry.getKey()) //
046                        .append("=$") //
047                        .append(entry.getKey()) //
048                        .append("\n");
049                }
050                final IRubyObject result = runtime.evalScriptlet(localVars + expression);
051                return JavaEmbedUtils.rubyToJava(runtime, result, Object.class);
052            } catch (final RuntimeException ex) {
053                throw new ExpressionEvaluationException("Evaluating JRuby expression failed: " + expression, ex);
054            }
055        }
056    }
