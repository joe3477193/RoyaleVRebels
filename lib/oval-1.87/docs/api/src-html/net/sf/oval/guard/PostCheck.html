013    package net.sf.oval.guard;
014    
015    import net.sf.oval.AbstractCheck;
016    import net.sf.oval.Validator;
017    import net.sf.oval.context.OValContext;
018    import net.sf.oval.exception.OValException;
019    
020    /**
021     * @author Sebastian Thomschke
022     */
023    public class PostCheck extends AbstractCheck {
024        private static final long serialVersionUID = 1L;
025    
026        private String expression;
027        private String language;
028        private String old;
029    
030        public void configure(final Post constraintAnnotation) {
031            setMessage(constraintAnnotation.message());
032            setErrorCode(constraintAnnotation.errorCode());
033            setSeverity(constraintAnnotation.severity());
034            setExpression(constraintAnnotation.expr());
035            setLanguage(constraintAnnotation.lang());
036            setOld(constraintAnnotation.old());
037            setProfiles(constraintAnnotation.profiles());
038        }
039    
040        public String getExpression() {
041            return expression;
042        }
043    
044        public String getLanguage() {
045            return language;
046        }
047    
048        public String getOld() {
049            return old;
050        }
051    
052        public boolean isSatisfied(final Object validatedObject, final Object valueToValidate, final OValContext context, final Validator validator)
053                throws OValException {
054            throw new UnsupportedOperationException();
055        }
056    
057        public void setExpression(final String condition) {
058            expression = condition;
059        }
060    
061        public void setLanguage(final String language) {
062            this.language = language;
063        }
064    
065        public void setOld(final String old) {
066            this.old = old;
067        }
068    }
