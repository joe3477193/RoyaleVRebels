013    package net.sf.oval.internal.util;

015    import java.lang.reflect.Method;
016    import java.util.ArrayList;

018    import java.util.Collections;

021    /**
022     * @author Sebastian Thomschke
023     */
024    public final class ArrayUtils {
025        public static final Method[] EMPTY_METHOD_ARRAY = {};
026        public static final Object[] EMPTY_OBJECT_ARRAY = {};
027        public static final String[] EMPTY_STRING_ARRAY = {};
028    
029        /**
030         * @throws IllegalArgumentException if <code>collection == null</code>
031         */
032        public static <T> int addAll(final Collection<T> collection, final T... elements) throws IllegalArgumentException {
033            if (elements == null)
034                return 0;
035    
036            int count = 0;
037            for (final T elem : elements)
038                if (collection.add(elem))
039                    count++;
040            return count;
041        }
042    
043        public static List<?> asList(final Object array) {
044            if (array instanceof Object[]) {
045                final Object[] arrayCasted = (Object[]) array;
046                final List<Object> result = new ArrayList<Object>(arrayCasted.length);
047                Collections.addAll(result, arrayCasted);
048                return result;
049            }
050            if (array instanceof byte[]) {
051                final byte[] arrayCasted = (byte[]) array;
052                final List<Byte> result = new ArrayList<Byte>(arrayCasted.length);
053                for (final byte i : arrayCasted)
054                    result.add(i);
055                return result;
056            }
057            if (array instanceof char[]) {
058                final char[] arrayCasted = (char[]) array;
059                final List<Character> result = new ArrayList<Character>(arrayCasted.length);
060                for (final char i : arrayCasted)
061                    result.add(i);
062                return result;
063            }
064            if (array instanceof short[]) {
065                final short[] arrayCasted = (short[]) array;
066                final List<Short> result = new ArrayList<Short>(arrayCasted.length);
067                for (final short i : arrayCasted)
068                    result.add(i);
069                return result;
070            }
071            if (array instanceof int[]) {
072                final int[] arrayCasted = (int[]) array;
073                final List<Integer> result = new ArrayList<Integer>(arrayCasted.length);
074                for (final int i : arrayCasted)
075                    result.add(i);
076                return result;
077            }
078            if (array instanceof long[]) {
079                final long[] arrayCasted = (long[]) array;
080                final List<Long> result = new ArrayList<Long>(arrayCasted.length);
081                for (final long i : arrayCasted)
082                    result.add(i);
083                return result;
084            }
085            if (array instanceof double[]) {
086                final double[] arrayCasted = (double[]) array;
087                final List<Double> result = new ArrayList<Double>(arrayCasted.length);
088                for (final double i : arrayCasted)
089                    result.add(i);
090                return result;
091            }
092            if (array instanceof float[]) {
093                final float[] arrayCasted = (float[]) array;
094                final List<Float> result = new ArrayList<Float>(arrayCasted.length);
095                for (final float i : arrayCasted)
096                    result.add(i);
097                return result;
098            }
099            if (array instanceof boolean[]) {
100                final boolean[] arrayCasted = (boolean[]) array;
101                final List<Boolean> result = new ArrayList<Boolean>(arrayCasted.length);
102                for (final boolean i : arrayCasted)
103                    result.add(i);
104                return result;
105            }
106    
107            throw new IllegalArgumentException("Argument [array] must be an array");
108        }
109    
110        public static <T> List<T> asList(final T[] array) {
111            final List<T> result = new ArrayList<T>(array.length);
112            Collections.addAll(result, array);
113            return result;
114        }
115    
116        public static <T> boolean containsEqual(final T[] theArray, final T theItem) {
117            for (final T t : theArray) {
118                if (t == theItem)
119                    return true;
120                if (t != null && t.equals(theItem))
121                    return true;
122            }
123            return false;
124        }
125    
126        public static <T> boolean containsSame(final T[] theArray, final T theItem) {
127            for (final T t : theArray)
128                if (t == theItem)
129                    return true;
130            return false;
131        }
132    
133        private ArrayUtils() {
134            super();
135        }
