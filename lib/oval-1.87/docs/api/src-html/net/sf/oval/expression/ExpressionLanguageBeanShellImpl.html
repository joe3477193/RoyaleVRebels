013    package net.sf.oval.expression;
014    
015    import java.util.Map;
016    import java.util.Map.Entry;
017    
018    import net.sf.oval.exception.ExpressionEvaluationException;
019    import net.sf.oval.internal.Log;
020    import bsh.EvalError;
021    import bsh.Interpreter;
022    
023    /**
024     * @author Sebastian Thomschke
025     */
026    public class ExpressionLanguageBeanShellImpl extends AbstractExpressionLanguage {
027        private static final Log LOG = Log.getLog(ExpressionLanguageBeanShellImpl.class);
028    
029        public Object evaluate(final String expression, final Map<String, ?> values) throws ExpressionEvaluationException {
030            LOG.debug("Evaluating BeanShell expression: {1}", expression);
031            try {
032                final Interpreter interpreter = new Interpreter();
033                interpreter.eval("setAccessibility(true)"); // turn off access restrictions
034                for (final Entry<String, ?> entry : values.entrySet()) {
035                    interpreter.set(entry.getKey(), entry.getValue());
036                }
037                return interpreter.eval(expression);
038            } catch (final EvalError ex) {
039                throw new ExpressionEvaluationException("Evaluating BeanShell expression failed: " + expression, ex);
040            }
041        }
042    }
