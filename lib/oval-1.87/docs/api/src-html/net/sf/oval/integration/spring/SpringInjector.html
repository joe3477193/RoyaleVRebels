013    package net.sf.oval.integration.spring;
014    
015    import org.springframework.beans.factory.annotation.Autowired;
016    import org.springframework.beans.factory.config.AutowireCapableBeanFactory;
017    import org.springframework.stereotype.Component;
018    import org.springframework.util.Assert;
019    
020    import net.sf.oval.internal.Log;
021    
022    /**
023     * Injects spring beans into unmanaged Java objects having {@link org.springframework.beans.factory.annotation.Autowired},
024     * {@link org.springframework.beans.factory.annotation.Value} and {@link javax.inject.Inject} annotations
025     * and executes {@link org.springframework.beans.factory.InitializingBean#afterPropertiesSet} and {@link javax.annotation.PostConstruct} callback methods.
026     *
027     * <pre>
028     * &lt;bean class="net.sf.oval.integration.spring.SpringInjector" /&gt;
029     * </pre>
030     *
031     * or
032     *
033     * <pre>
034     * &lt;context:component-scan base-package="net.sf.oval.integration.spring" /&gt;
035     * </pre>
036     *
037     * @author Sebastian Thomschke
038     */
039    @Component
040    public class SpringInjector {
041        private static final Log LOG = Log.getLog(SpringInjector.class);
042    
043        private static SpringInjector INSTANCE;
044    
045        public static SpringInjector get() {
046            Assert.notNull(INSTANCE, "No SpringInjector instance created yet. Add  <bean class=\"" + SpringInjector.class.getName()
047                    + "\" /> to your spring configuration!");
048    
049            return INSTANCE;
050        }
051    
052        @Autowired
053        private AutowireCapableBeanFactory beanFactory;
054    
055        private SpringInjector() {
056            LOG.info("Instantiated.");
057    
058            INSTANCE = this;
059        }
060    
061        /**
062         * processes @PostConstruct, InitializingBean#afterPropertiesSet
063         */
064        public void initialize(final Object unmanagedBean) {
065            beanFactory.initializeBean(unmanagedBean, "bean");
066        }
067    
068        /**
069         * processes @Autowired, @Inject
070         */
071        public void inject(final Object unmanagedBean) {
072            beanFactory.autowireBean(unmanagedBean);
073        }
074    }
