052    /**
053     * Constraints configurer that interprets certain EJB3 JPA annotations:
054     * 
055     * <pre>
056     * * javax.persistence.Basic(optional=false)     => net.sf.oval.constraint.NotNullCheck
057     * * javax.persistence.OneToOne(optional=false)  => net.sf.oval.constraint.NotNullCheck, net.sf.oval.constraint.AssertValidCheck (if addAssertValidConstraints=true)
058     * * javax.persistence.ManyToOne(optional=false) => net.sf.oval.constraint.NotNullCheck, net.sf.oval.constraint.AssertValidCheck (if addAssertValidConstraints=true)
059     * * javax.persistence.ManyToMany                => net.sf.oval.constraint.AssertValidCheck (if addAssertValidConstraints=true)
060     * * javax.persistence.Column(nullable=false)    => net.sf.oval.constraint.NotNullCheck
061     * * javax.persistence.Column(length=5)          => net.sf.oval.constraint.LengthCheck
062     * * javax.persistence.Column(precision>0)       => net.sf.oval.constraint.RangeCheck (for Numbers only)
063     * </pre>
064     *
065     * <b>Important:</b> by default AssertValidChecks are added for n-m relationships. This may be a problem when using lazy loading. Read
066     * <a href="http://sourceforge.net/p/oval/discussion/488110/thread/6ec11584/#4ae0">this post</a> for more details.
067     * To avoid this override the method {@link #addAssertValidCheckIfRequired(Annotation, Collection, AccessibleObject)} with an empty method body, for example
068     * 
069     * <pre>
070     * JPAAnnotationsConfigurer configurer = new JPAAnnotationsConfigurer() {
071     *     protected void addAssertValidCheckIfRequired(Annotation constraintAnnotation, Collection<Check> checks, AccessibleObject fieldOrMethod) {
072     *         // do nothing
073     *     }
074     * };
075     * </pre>
076     *
077     * @author Sebastian Thomschke
078     */
079    @SuppressWarnings("javadoc")
080    public class JPAAnnotationsConfigurer implements Configurer {
081        protected Boolean applyFieldConstraintsToSetters;
082        protected Boolean applyFieldConstraintsToConstructors;
083    
084        protected void addAssertValidCheckIfRequired(final Annotation constraintAnnotation, final Collection<Check> checks,
085                @SuppressWarnings("unused") /*parameter for potential use by subclasses*/final AccessibleObject fieldOrMethod) {
086            if (containsCheckOfType(checks, AssertValidCheck.class))
087                return;
088    
089            if (constraintAnnotation instanceof OneToOne || constraintAnnotation instanceof OneToMany || constraintAnnotation instanceof ManyToOne
090                    || constraintAnnotation instanceof ManyToMany)
091                checks.add(new AssertValidCheck());
092        }
093    
094        protected boolean containsCheckOfType(final Collection<Check> checks, final Class<? extends Check> checkClass) {
095            for (final Check check : checks)
096                if (checkClass.isInstance(check))
097                    return true;
098            return false;
099        }
100    
101        public Boolean getApplyFieldConstraintsToConstructors() {
102            return applyFieldConstraintsToConstructors;
103        }
104    
105        public ClassConfiguration getClassConfiguration(final Class<?> clazz) {
106            final CollectionFactory cf = getCollectionFactory();
107    
108            final ClassConfiguration config = new ClassConfiguration();
109            config.type = clazz;
110            config.applyFieldConstraintsToConstructors = applyFieldConstraintsToConstructors;
111            config.applyFieldConstraintsToSetters = applyFieldConstraintsToSetters;
112    
113            List<Check> checks = cf.createList(2);
114    
115            /*
116             * determine field checks
117             */
118            for (final Field field : config.type.getDeclaredFields()) {
119    
120                // loop over all annotations of the current field
121                for (final Annotation annotation : field.getAnnotations()) {
122                    if (annotation instanceof Basic)
123                        initializeChecks((Basic) annotation, checks);
124                    else if (annotation instanceof Column)
125                        initializeChecks((Column) annotation, checks, field);
126                    else if (annotation instanceof OneToOne)
127                        initializeChecks((OneToOne) annotation, checks);
128                    else if (annotation instanceof ManyToOne)
129                        initializeChecks((ManyToOne) annotation, checks);
130                    else if (annotation instanceof ManyToMany)
131                        initializeChecks((ManyToMany) annotation, checks);
132                    else if (annotation instanceof OneToMany)
133                        initializeChecks((OneToMany) annotation, checks);
134    
135                    addAssertValidCheckIfRequired(annotation, checks, field);
136                }
137    
138                if (checks.size() > 0) {
139                    if (config.fieldConfigurations == null)
140                        config.fieldConfigurations = cf.createSet(8);
141    
142                    final FieldConfiguration fc = new FieldConfiguration();
143                    fc.name = field.getName();
144                    fc.checks = checks;
145                    checks = cf.createList(); // create a new list for the next field with checks
146                    config.fieldConfigurations.add(fc);
147                }
148            }
149    
150            /*
151             * determine getter checks
152             */
153            for (final Method method : config.type.getDeclaredMethods()) {
154                // consider getters only
155                if (!ReflectionUtils.isGetter(method))
156                    continue;
157    
158                // loop over all annotations
159                for (final Annotation annotation : method.getAnnotations()) {
160                    if (annotation instanceof Basic)
161                        initializeChecks((Basic) annotation, checks);
162                    else if (annotation instanceof Column)
163                        initializeChecks((Column) annotation, checks, method);
164                    else if (annotation instanceof OneToOne)
165                        initializeChecks((OneToOne) annotation, checks);
166                    else if (annotation instanceof ManyToOne)
167                        initializeChecks((ManyToOne) annotation, checks);
168                    else if (annotation instanceof ManyToMany)
169                        initializeChecks((ManyToMany) annotation, checks);
170                    else if (annotation instanceof OneToMany)
171                        initializeChecks((OneToMany) annotation, checks);
172    
173                    addAssertValidCheckIfRequired(annotation, checks, method);
174                }
175    
176                // check if anything has been configured for this method at all
177                if (checks.size() > 0) {
178                    if (config.methodConfigurations == null)
179                        config.methodConfigurations = cf.createSet(2);
180    
181                    final MethodConfiguration mc = new MethodConfiguration();
182                    mc.name = method.getName();
183                    mc.isInvariant = true;
184                    mc.returnValueConfiguration = new MethodReturnValueConfiguration();
185                    mc.returnValueConfiguration.checks = checks;
186                    checks = cf.createList(); // create a new list for the next method having return value checks
187                    config.methodConfigurations.add(mc);
188                }
189            }
190            return config;
191        }

193        public ConstraintSetConfiguration getConstraintSetConfiguration(final String constraintSetId) {
194            return null;
195        }
196    
197        protected void initializeChecks(final Basic annotation, final Collection<Check> checks) {
198            if (!annotation.optional() && !containsCheckOfType(checks, NotNullCheck.class))
199                checks.add(new NotNullCheck());
200        }
201    
202        protected void initializeChecks(final Column annotation, final Collection<Check> checks, final AccessibleObject fieldOrMethod) {
203            /* If the value is generated (annotated with @GeneratedValue) it is allowed to be null
204             * before the entity has been persisted, same is true in case of optimistic locking
205             * when a field is annotated with @Version.
206             * Therefore and because of the fact that there is no generic way to determine if an entity
207             * has been persisted already, a not-null check will not be performed for such fields.
208             */
209            if (!annotation.nullable() && !fieldOrMethod.isAnnotationPresent(GeneratedValue.class) && !fieldOrMethod.isAnnotationPresent(Version.class)
210                    && !fieldOrMethod.isAnnotationPresent(NotNull.class))
211                if (!containsCheckOfType(checks, NotNullCheck.class))
212                    checks.add(new NotNullCheck());
213    
214            // add Length check based on Column.length parameter, but only:
215            if (!fieldOrMethod.isAnnotationPresent(Lob.class) && // if @Lob is not present
216                    !fieldOrMethod.isAnnotationPresent(Enumerated.class) && // if @Enumerated is not present
217                    !fieldOrMethod.isAnnotationPresent(Length.class) // if an explicit @Length constraint is not present
218            ) {
219                final LengthCheck lengthCheck = new LengthCheck();
220                lengthCheck.setMax(annotation.length());
221                checks.add(lengthCheck);
222            }
223    
224            // add Range check based on Column.precision/scale parameters, but only:
225            if (!fieldOrMethod.isAnnotationPresent(Range.class) // if an explicit @Range is not present
226                    && annotation.precision() > 0 // if precision is > 0
227                    && Number.class.isAssignableFrom(fieldOrMethod instanceof Field ? ((Field) fieldOrMethod).getType() : ((Method) fieldOrMethod).getReturnType()) // if numeric field type
228            ) {
229                /* precision = 6, scale = 2  => -9999.99<=x<=9999.99
230                 * precision = 4, scale = 1  =>   -999.9<=x<=999.9
231                 */
232                final RangeCheck rangeCheck = new RangeCheck();
233                rangeCheck.setMax(Math.pow(10, annotation.precision() - annotation.scale()) - Math.pow(0.1, annotation.scale()));
234                rangeCheck.setMin(-1 * rangeCheck.getMax());
235                checks.add(rangeCheck);
236            }
237        }

240        protected void initializeChecks(final ManyToMany annotation, final Collection<Check> checks) {
241            // override if required
242        }
243    
244        protected void initializeChecks(final ManyToOne annotation, final Collection<Check> checks) {
245            if (!annotation.optional() && !containsCheckOfType(checks, NotNullCheck.class))
246                checks.add(new NotNullCheck());
247        }

250        protected void initializeChecks(final OneToMany annotation, final Collection<Check> checks) {
251            // override if required
252        }
253    
254        protected void initializeChecks(final OneToOne annotation, final Collection<Check> checks) {
255            if (!annotation.optional() && !containsCheckOfType(checks, NotNullCheck.class))
256                checks.add(new NotNullCheck());
257        }
