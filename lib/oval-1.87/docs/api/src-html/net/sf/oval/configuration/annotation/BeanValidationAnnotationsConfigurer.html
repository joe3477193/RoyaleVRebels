066    /**
067     * Constraints configurer that interprets the JSR303 built-in Java Bean Validation annotations:
068     * <ul>
069     * <li>javax.validation.constraints.AssertFalse => net.sf.oval.constraint.AssertFalseCheck
070     * <li>javax.validation.constraints.AssertTrue => net.sf.oval.constraint.AssertTrueCheck
071     * <li>javax.validation.constraints.DecimalMax => net.sf.oval.constraint.MaxCheck
072     * <li>javax.validation.constraints.DecimalMin => net.sf.oval.constraint.MinCheck
073     * <li>javax.validation.constraints.Digits => net.sf.oval.constraint.DigitsCheck
074     * <li>javax.validation.constraints.Future => net.sf.oval.constraint.FutureCheck
075     * <li>javax.validation.constraints.Max => net.sf.oval.constraint.MaxCheck
076     * <li>javax.validation.constraints.Min => net.sf.oval.constraint.MinCheck
077     * <li>javax.validation.constraints.NotNull => net.sf.oval.constraint.NotNullCheck
078     * <li>javax.validation.constraints.Null => net.sf.oval.constraint.AssertNullCheck
079     * <li>javax.validation.constraints.Past => net.sf.oval.constraint.PastCheck
080     * <li>javax.validation.constraints.Pattern => net.sf.oval.constraint.PatternCheck
081     * <li>javax.validation.constraints.Size => net.sf.oval.constraint.SizeCheck
082     * <li>javax.validation.Valid => net.sf.oval.constraint.AssertValidCheck
083     * </ul>
084     *
085     * @author Sebastian Thomschke
086     */
087    public class BeanValidationAnnotationsConfigurer implements Configurer {
088        private static final Log LOG = Log.getLog(BeanValidationAnnotationsConfigurer.class);
089    
090        private List<ParameterConfiguration> _createParameterConfiguration(final Annotation[][] paramAnnotations, final Class<?>[] parameterTypes) {
091            final CollectionFactory cf = getCollectionFactory();
092    
093            final List<ParameterConfiguration> paramCfg = cf.createList(paramAnnotations.length);
094    
095            List<Check> paramChecks = cf.createList(2);
096    
097            // loop over all parameters of the current constructor
098            for (int i = 0; i < paramAnnotations.length; i++) {
099                // loop over all annotations of the current constructor parameter
100                for (final Annotation annotation : paramAnnotations[i]) {
101                    initializeChecks(annotation, paramChecks);
102                }
103    
104                final ParameterConfiguration pc = new ParameterConfiguration();
105                paramCfg.add(pc);
106                pc.type = parameterTypes[i];
107                if (paramChecks.size() > 0) {
108                    pc.checks = paramChecks;
109                    paramChecks = cf.createList(2); // create a new list for the next parameter having checks
110                }
111            }
112            return paramCfg;
113        }
114    
115        protected void configureConstructorParameterChecks(final ClassConfiguration classCfg) {
116            final CollectionFactory cf = getCollectionFactory();
117    
118            for (final Constructor<?> ctor : classCfg.type.getDeclaredConstructors()) {
119                final List<ParameterConfiguration> paramCfg = _createParameterConfiguration(ctor.getParameterAnnotations(), ctor.getParameterTypes());
120    
121                if (paramCfg.size() > 0) {
122                    if (classCfg.constructorConfigurations == null) {
123                        classCfg.constructorConfigurations = cf.createSet(2);
124                    }
125    
126                    final ConstructorConfiguration cc = new ConstructorConfiguration();
127                    cc.parameterConfigurations = paramCfg;
128                    cc.postCheckInvariants = false;
129                    classCfg.constructorConfigurations.add(cc);
130                }
131            }
132        }
133    
134        protected void configureFieldChecks(final ClassConfiguration classCfg) {
135            final CollectionFactory cf = getCollectionFactory();
136    
137            List<Check> checks = cf.createList(2);
138    
139            for (final Field field : classCfg.type.getDeclaredFields()) {
140                // loop over all annotations of the current field
141                for (final Annotation annotation : field.getAnnotations()) {
142                    initializeChecks(annotation, checks);
143                }
144    
145                if (checks.size() > 0) {
146                    if (classCfg.fieldConfigurations == null) {
147                        classCfg.fieldConfigurations = cf.createSet(2);
148                    }
149    
150                    final FieldConfiguration fc = new FieldConfiguration();
151                    fc.name = field.getName();
152                    fc.checks = checks;
153                    classCfg.fieldConfigurations.add(fc);
154                    checks = cf.createList(2); // create a new list for the next field with checks
155                }
156            }
157        }
158    
159        /**
160         * configure method return value and parameter checks
161         */
162        protected void configureMethodChecks(final ClassConfiguration classCfg) {
163            final CollectionFactory cf = getCollectionFactory();
164    
165            List<Check> returnValueChecks = cf.createList(2);
166    
167            for (final Method method : classCfg.type.getDeclaredMethods()) {
168                // loop over all annotations
169                for (final Annotation annotation : ReflectionUtils.getAnnotations(method, Boolean.TRUE.equals(classCfg.inspectInterfaces))) {
170                    initializeChecks(annotation, returnValueChecks);
171                }
172    
173                /*
174                 * determine parameter checks
175                 */
176                final List<ParameterConfiguration> paramCfg = _createParameterConfiguration(ReflectionUtils.getParameterAnnotations(method, Boolean.TRUE.equals(
177                    classCfg.inspectInterfaces)), method.getParameterTypes());
178    
179                // check if anything has been configured for this method at all
180                if (paramCfg.size() > 0 || returnValueChecks.size() > 0) {
181                    if (classCfg.methodConfigurations == null) {
182                        classCfg.methodConfigurations = cf.createSet(2);
183                    }
184    
185                    final MethodConfiguration mc = new MethodConfiguration();
186                    mc.name = method.getName();
187                    mc.parameterConfigurations = paramCfg;
188                    mc.isInvariant = ReflectionUtils.isGetter(method);
189                    if (returnValueChecks.size() > 0) {
190                        mc.returnValueConfiguration = new MethodReturnValueConfiguration();
191                        mc.returnValueConfiguration.checks = returnValueChecks;
192                        returnValueChecks = cf.createList(2); // create a new list for the next method having return value checks
193                    }
194                    classCfg.methodConfigurations.add(mc);
195                }
196            }
197        }

199        public ClassConfiguration getClassConfiguration(final Class<?> clazz) {
200            final ClassConfiguration classCfg = new ClassConfiguration();
201            classCfg.type = clazz;
202    
203            final Guarded guarded = clazz.getAnnotation(Guarded.class);
204    
205            if (guarded == null) {
206                classCfg.applyFieldConstraintsToConstructors = false;
207                classCfg.applyFieldConstraintsToSetters = false;
208                classCfg.assertParametersNotNull = false;
209                classCfg.checkInvariants = false;
210                classCfg.inspectInterfaces = false;
211            } else {
212                classCfg.applyFieldConstraintsToConstructors = guarded.applyFieldConstraintsToConstructors();
213                classCfg.applyFieldConstraintsToSetters = guarded.applyFieldConstraintsToSetters();
214                classCfg.assertParametersNotNull = guarded.assertParametersNotNull();
215                classCfg.checkInvariants = guarded.checkInvariants();
216                classCfg.inspectInterfaces = guarded.inspectInterfaces();
217            }
218    
219            configureFieldChecks(classCfg);
220            configureConstructorParameterChecks(classCfg);
221            configureMethodChecks(classCfg);
222    
223            return classCfg;
224        }

230        protected void initializeChecks(final Annotation annotation, final Collection<Check> checks) {
231            assert annotation != null;
232            assert checks != null;
233    
234            // ignore non-bean validation annotations
235            if (!(annotation instanceof Valid) && annotation.annotationType().getAnnotation(javax.validation.Constraint.class) == null)
236                return;
237    
238            Class<?>[] groups = null;
239            Check check = null;
240            if (annotation instanceof NotNull) {
241                groups = ((NotNull) annotation).groups();
242                check = new NotNullCheck();
243            } else if (annotation instanceof Null) {
244                groups = ((Null) annotation).groups();
245                check = new AssertNullCheck();
246            } else if (annotation instanceof Valid) {
247                check = new AssertValidCheck();
248            } else if (annotation instanceof AssertTrue) {
249                groups = ((AssertTrue) annotation).groups();
250                check = new AssertTrueCheck();
251            } else if (annotation instanceof AssertFalse) {
252                groups = ((AssertFalse) annotation).groups();
253                check = new AssertFalseCheck();
254            } else if (annotation instanceof DecimalMax) {
255                groups = ((DecimalMax) annotation).groups();
256                final MaxCheck maxCheck = new MaxCheck();
257                maxCheck.setMax(Double.parseDouble(((DecimalMax) annotation).value()));
258                check = maxCheck;
259            } else if (annotation instanceof DecimalMin) {
260                groups = ((DecimalMin) annotation).groups();
261                final MinCheck minCheck = new MinCheck();
262                minCheck.setMin(Double.parseDouble(((DecimalMin) annotation).value()));
263                check = minCheck;
264            } else if (annotation instanceof Max) {
265                groups = ((Max) annotation).groups();
266                final MaxCheck maxCheck = new MaxCheck();
267                maxCheck.setMax(((Max) annotation).value());
268                check = maxCheck;
269            } else if (annotation instanceof Min) {
270                groups = ((Min) annotation).groups();
271                final MinCheck minCheck = new MinCheck();
272                minCheck.setMin(((Min) annotation).value());
273                check = minCheck;
274            } else if (annotation instanceof Future) {
275                groups = ((Future) annotation).groups();
276                check = new FutureCheck();
277            } else if (annotation instanceof Past) {
278                groups = ((Past) annotation).groups();
279                check = new PastCheck();
280            } else if (annotation instanceof Pattern) {
281                groups = ((Pattern) annotation).groups();
282                final MatchPatternCheck matchPatternCheck = new MatchPatternCheck();
283                int iflag = 0;
284                for (final Flag flag : ((Pattern) annotation).flags()) {
285                    iflag = iflag | flag.getValue();
286                }
287                matchPatternCheck.setPattern(((Pattern) annotation).regexp(), iflag);
288                check = matchPatternCheck;
289            } else if (annotation instanceof Digits) {
290                groups = ((Digits) annotation).groups();
291                final DigitsCheck digitsCheck = new DigitsCheck();
292                digitsCheck.setMaxFraction(((Digits) annotation).fraction());
293                digitsCheck.setMaxInteger(((Digits) annotation).integer());
294                check = digitsCheck;
295            } else if (annotation instanceof Size) {
296                groups = ((Size) annotation).groups();
297                final SizeCheck sizeCheck = new SizeCheck();
298                sizeCheck.setMax(((Size) annotation).max());
299                sizeCheck.setMin(((Size) annotation).min());
300                check = sizeCheck;
301            }
302    
303            if (check != null) {
304                final Method getMessage = ReflectionUtils.getMethod(annotation.getClass(), "message", (Class<?>[]) null);
305                if (getMessage != null) {
306                    final String message = ReflectionUtils.invokeMethod(getMessage, annotation, (Object[]) null);
307                    if (message != null && !message.startsWith("{javax.validation.constraints.")) {
308                        check.setMessage(message);
309                    }
310                }
311    
312                if (groups != null && groups.length > 0) {
313                    final String[] profiles = new String[groups.length];
314                    for (int i = 0, l = groups.length; i < l; i++) {
315                        profiles[i] = groups[i].getName();
316                    }
317                    check.setProfiles(profiles);
318                }
319                checks.add(check);
320                return;
321            }
322    
323            Annotation[] list = null;
324            if (annotation instanceof AssertFalse.List) {
325                list = ((AssertFalse.List) annotation).value();
326            } else if (annotation instanceof AssertTrue.List) {
327                list = ((AssertTrue.List) annotation).value();
328            } else if (annotation instanceof DecimalMax.List) {
329                list = ((DecimalMax.List) annotation).value();
330            } else if (annotation instanceof DecimalMin.List) {
331                list = ((DecimalMin.List) annotation).value();
332            } else if (annotation instanceof Digits.List) {
333                list = ((Digits.List) annotation).value();
334            } else if (annotation instanceof Future.List) {
335                list = ((Future.List) annotation).value();
336            } else if (annotation instanceof Max.List) {
337                list = ((Max.List) annotation).value();
338            } else if (annotation instanceof Min.List) {
339                list = ((Min.List) annotation).value();
340            } else if (annotation instanceof NotNull.List) {
341                list = ((NotNull.List) annotation).value();
342            } else if (annotation instanceof Null.List) {
343                list = ((Null.List) annotation).value();
344            } else if (annotation instanceof Past.List) {
345                list = ((Past.List) annotation).value();
346            } else if (annotation instanceof Pattern.List) {
347                list = ((Pattern.List) annotation).value();
348            } else if (annotation instanceof Size.List) {
349                list = ((Size.List) annotation).value();
350            }
351    
352            if (list != null) {
353                for (final Annotation anno : list) {
354                    initializeChecks(anno, checks);
355                }
356            } else {
357                LOG.warn("Ignoring unsupported bean validation constraint annotation {1}", annotation);
358                return;
359            }
360        }
