013    package net.sf.oval.ogn;
014    
015    import java.lang.reflect.AccessibleObject;
016    import java.lang.reflect.Field;
017    import java.lang.reflect.Method;
018    
019    import net.sf.oval.exception.InvalidConfigurationException;
020    import net.sf.oval.internal.util.Assert;
021    import net.sf.oval.internal.util.ReflectionUtils;
022    
023    /**
024     * Default object graph navigator implementation.
025     * 
026     * Object path separator is a colon (.), e.g. owner.address.street
027     * 
028     * The implementation currently is limited to address fields and properties. Separate items of arrays, maps or keys cannot be addressed.
029     * 
030     * @author Sebastian Thomschke
031     */
032    public class ObjectGraphNavigatorDefaultImpl implements ObjectGraphNavigator {
033        public ObjectGraphNavigationResult navigateTo(final Object root, final String path) throws InvalidConfigurationException {
034            Assert.argumentNotNull("root", root);
035            Assert.argumentNotNull("path", path);
036    
037            Object parent = null;
038            Object target = root;
039            AccessibleObject targetAccessor = null;
040            for (final String chunk : path.split("\\.")) {
041                parent = target;
042                if (parent == null)
043                    return null;
044                final Field field = ReflectionUtils.getFieldRecursive(parent.getClass(), chunk);
045                if (field == null) {
046                    final Method getter = ReflectionUtils.getGetterRecursive(parent.getClass(), chunk);
047                    if (getter == null)
048                        throw new InvalidConfigurationException("Invalid object navigation path from root object class [" + root.getClass().getName() + "] path: "
049                                + path);
050                    targetAccessor = getter;
051                    target = ReflectionUtils.invokeMethod(getter, parent);
052                } else {
053                    targetAccessor = field;
054                    target = ReflectionUtils.getFieldValue(field, parent);
055                }
056            }
057            return new ObjectGraphNavigationResult(root, path, parent, targetAccessor, target);
058        }
059    }
