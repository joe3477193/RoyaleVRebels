013    package net.sf.oval.collection;
014    
015    import java.util.ArrayList;
016    import java.util.List;
017    import java.util.Map;
018    import java.util.Set;
019    
020    import gnu.trove.map.hash.THashMap;
021    import gnu.trove.set.hash.THashSet;
022    
023    /**
024     * @author Sebastian Thomschke
025     */
026    public class CollectionFactoryTroveImpl implements CollectionFactory {
027    
028        public <ValueType> List<ValueType> createList() {
029            return new ArrayList<ValueType>();
030        }
031    
032        public <ValueType> List<ValueType> createList(final int initialCapacity) {
033            return new ArrayList<ValueType>(initialCapacity);
034        }
035    
036        public <KeyType, ValueType> Map<KeyType, ValueType> createMap() {
037            return new THashMap<KeyType, ValueType>();
038        }
039    
040        public <KeyType, ValueType> Map<KeyType, ValueType> createMap(final int initialCapacity) {
041            return new THashMap<KeyType, ValueType>(initialCapacity);
042        }
043    
044        public <ValueType> Set<ValueType> createSet() {
045            return new THashSet<ValueType>();
046        }
047    
048        public <ValueType> Set<ValueType> createSet(final int initialCapacity) {
049            return new THashSet<ValueType>(initialCapacity);
050        }
051    }
