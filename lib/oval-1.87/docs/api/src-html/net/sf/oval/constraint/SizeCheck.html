013    package net.sf.oval.constraint;
014    
015    import static net.sf.oval.Validator.*;
016    
017    import java.lang.reflect.Array;
018    import java.util.Collection;
019    import java.util.Map;
020    
021    import net.sf.oval.Validator;
022    import net.sf.oval.configuration.annotation.AbstractAnnotationCheck;
023    import net.sf.oval.context.OValContext;
024    
025    /**
026     * @author Sebastian Thomschke
027     */
028    public class SizeCheck extends AbstractAnnotationCheck<Size> {
029        private static final long serialVersionUID = 1L;
030    
031        private int min;
032        private int max;
033    
034        @Override
035        public void configure(final Size constraintAnnotation) {
036            super.configure(constraintAnnotation);
037            setMax(constraintAnnotation.max());
038            setMin(constraintAnnotation.min());
039        }
040    
041        @Override
042        protected Map<String, String> createMessageVariables() {
043            final Map<String, String> messageVariables = getCollectionFactory().createMap(2);
044            messageVariables.put("max", Integer.toString(max));
045            messageVariables.put("min", Integer.toString(min));
046            return messageVariables;
047        }
048    
049        public int getMax() {
050            return max;
051        }
052    
053        public int getMin() {
054            return min;
055        }
056    
057        public boolean isSatisfied(final Object validatedObject, final Object valueToValidate, final OValContext context, final Validator validator) {
058            if (valueToValidate == null)
059                return true;
060    
061            if (valueToValidate instanceof Collection<?>) {
062                final int size = ((Collection<?>) valueToValidate).size();
063                return size >= min && size <= max;
064            }
065            if (valueToValidate instanceof Map<?, ?>) {
066                final int size = ((Map<?, ?>) valueToValidate).size();
067                return size >= min && size <= max;
068            }
069            if (valueToValidate.getClass().isArray()) {
070                final int size = Array.getLength(valueToValidate);
071                return size >= min && size <= max;
072            }
073            final String str = valueToValidate.toString();
074            final int size = str.length();
075            return size >= min && size <= max;
076        }
077    
078        public void setMax(final int max) {
079            this.max = max;
080            requireMessageVariablesRecreation();
081        }
082    
083        public void setMin(final int min) {
084            this.min = min;
085            requireMessageVariablesRecreation();
086        }
087    }
