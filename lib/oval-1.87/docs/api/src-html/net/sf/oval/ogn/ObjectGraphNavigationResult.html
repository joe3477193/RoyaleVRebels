013    package net.sf.oval.ogn;
014    
015    import java.lang.reflect.AccessibleObject;
016    
017    /**
018     * @author Sebastian Thomschke
019     */
020    public class ObjectGraphNavigationResult {
021        public final Object root;
022    
023        public final String path;
024    
025        public final Object targetParent;
026    
027        /**
028         * field or method
029         */
030        public final AccessibleObject targetAccessor;
031    
032        /**
033         * accessor's value
034         */
035        public final Object target;
036    
037        public ObjectGraphNavigationResult(final Object root, final String path, final Object targetParent, final AccessibleObject targetAccessor,
038                final Object target) {
039            this.root = root;
040            this.path = path;
041            this.targetParent = targetParent;
042            this.targetAccessor = targetAccessor;
043            this.target = target;
044        }
045    }
