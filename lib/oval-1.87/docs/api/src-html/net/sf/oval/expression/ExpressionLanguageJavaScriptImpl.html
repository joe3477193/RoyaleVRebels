013    package net.sf.oval.expression;
014    
015    import java.util.Map;
016    import java.util.Map.Entry;
017    
018    import net.sf.oval.exception.ExpressionEvaluationException;
019    import net.sf.oval.internal.Log;
020    import net.sf.oval.internal.util.ObjectCache;
021    
022    import org.mozilla.javascript.Context;
023    import org.mozilla.javascript.ContextFactory;
024    import org.mozilla.javascript.EvaluatorException;
025    import org.mozilla.javascript.Script;
026    import org.mozilla.javascript.Scriptable;
027    
028    /**
029     * @author Sebastian Thomschke
030     */
031    public class ExpressionLanguageJavaScriptImpl extends AbstractExpressionLanguage {
032        private static final Log LOG = Log.getLog(ExpressionLanguageJavaScriptImpl.class);
033    
034        private final Scriptable parentScope;
035    
036        private final ObjectCache<String, Script> scriptCache = new ObjectCache<String, Script>();
037    
038        public ExpressionLanguageJavaScriptImpl() {
039            final Context ctx = ContextFactory.getGlobal().enterContext();
040            try {
041                parentScope = ctx.initStandardObjects();
042            } finally {
043                Context.exit();
044            }
045        }
046    
047        public Object evaluate(final String expression, final Map<String, ?> values) throws ExpressionEvaluationException {
048            LOG.debug("Evaluating JavaScript expression: {1}", expression);
049            try {
050                final Context ctx = ContextFactory.getGlobal().enterContext();
051                Script script = scriptCache.get(expression);
052                if (script == null) {
053                    ctx.setOptimizationLevel(9);
054                    script = ctx.compileString(expression, "<cmd>", 1, null);
055                    scriptCache.put(expression, script);
056                }
057                final Scriptable scope = ctx.newObject(parentScope);
058                scope.setPrototype(parentScope);
059                scope.setParentScope(null);
060                for (final Entry<String, ?> entry : values.entrySet()) {
061                    scope.put(entry.getKey(), scope, Context.javaToJS(entry.getValue(), scope));
062                }
063                return script.exec(ctx, scope);
064            } catch (final EvaluatorException ex) {
065                throw new ExpressionEvaluationException("Evaluating JavaScript expression failed: " + expression, ex);
066            } finally {
067                Context.exit();
068            }
069        }
070    }
