37  /**
38   * @author Sebastian Thomschke
39   */
40  public class GuiceInjectorTest extends TestCase {
41      @Retention(RetentionPolicy.RUNTIME)
42      @Target({ ElementType.FIELD })
43      @Constraint(checkWith = GuiceNullContraintCheck.class)
44      public @interface GuiceNullContraint {
45          //nothing
46      }
47  
48      public static class Entity {
49          @GuiceNullContraint
50          protected String field;
51      }
52  
53      /**
54       * constraint check implementation requiring Guice injected members
55       */
56      public static class GuiceNullContraintCheck extends AbstractAnnotationCheck<GuiceNullContraint> {
57          private static final long serialVersionUID = 1L;
58  
59          @Inject
60          @Named("GUICE_MANAGED_OBJECT")
61          private Integer guiceManagedObject;
62  
63          public boolean isSatisfied(final Object validatedObject, final Object valueToValidate, final OValContext context, final Validator validator)
64                  throws OValException {
65              return guiceManagedObject == 10 && valueToValidate != null;
66          }
67      }
68  
69      public void testWithGuiceInjector() {
70          final Injector injector = Guice.createInjector(new Module() {
71              public void configure(final Binder binder) {
72                  binder.bind(Integer.class).annotatedWith(Names.named("GUICE_MANAGED_OBJECT")).toInstance(10);
73              }
74          });
75  
76          final AnnotationsConfigurer myConfigurer = new AnnotationsConfigurer();
77          myConfigurer.addCheckInitializationListener(new GuiceCheckInitializationListener(injector));
78          final Validator v = new Validator(myConfigurer);
79  
80          final Entity e = new Entity();
81          assertEquals(1, v.validate(e).size());
82          e.field = "whatever";
83          assertEquals(0, v.validate(e).size());
84      }
85  
86      public void testWithoutGuiceInjector() {
87          final Validator v = new Validator();
88          final Entity e = new Entity();
89          try {
90              v.validate(e);
91              fail("NPE expected.");
92          } catch (final NullPointerException ex) {}
93      }
94  
95  }
