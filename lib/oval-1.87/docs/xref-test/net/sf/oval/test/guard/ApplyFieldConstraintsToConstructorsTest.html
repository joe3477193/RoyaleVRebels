13  package net.sf.oval.test.guard;
14  
15  import junit.framework.TestCase;
16  import net.sf.oval.constraint.AssertTrue;
17  import net.sf.oval.constraint.Length;
18  import net.sf.oval.constraint.MatchPattern;
19  import net.sf.oval.constraint.NotEmpty;
20  import net.sf.oval.constraint.NotNull;
21  import net.sf.oval.exception.ConstraintsViolatedException;
22  import net.sf.oval.guard.Guarded;
23  
24  /**
25   * @author Sebastian Thomschke
26   */
27  public class ApplyFieldConstraintsToConstructorsTest extends TestCase {
28      @Guarded(applyFieldConstraintsToConstructors = true, checkInvariants = false)
29      private static class Person {
30          @AssertTrue(message = "ASSERT_TRUE")
31          private boolean isValid = true;
32  
33          @NotNull(message = "NOT_NULL")
34          private String firstName = "";
35  
36          @NotNull(message = "NOT_NULL")
37          private String lastName = "";
38  
39          @NotNull(message = "NOT_NULL")
40          @Length(max = 6, message = "LENGTH")
41          @NotEmpty(message = "NOT_EMPTY")
42          @MatchPattern(pattern = "^[0-9]*$", message = "REG_EX")
43          private String zipCode = "1";
44  
45          protected Person(final boolean isValid, final String firstName, final String lastName, final String zipCode) {
46              super();
47              this.isValid = isValid;
48              this.firstName = firstName;
49              this.lastName = lastName;
50              this.zipCode = zipCode;
51          }
52  
53          protected Person(final String theFirstName, final String theLastName, final String theZipCode) {
54              super();
55              firstName = theFirstName;
56              lastName = theLastName;
57              zipCode = theZipCode;
58          }
59      }
60  
61      /**
62       * by default constraints specified for a field are also used for validating
63       * method parameters of the corresponding setter methods
64       */
65      @SuppressWarnings("unused")
66      public void testConstrucorParameterValidation() {
67          try {
68              new Person(false, null, null, null);
69          } catch (final ConstraintsViolatedException ex) {
70              assertEquals(ex.getConstraintViolations().length, 4);
71          }
72  
73          try {
74              new Person(true, "", "", "12345");
75          } catch (final ConstraintsViolatedException ex) {
76              fail();
77          }
78  
79          try {
80              new Person(null, null, null);
81          } catch (final ConstraintsViolatedException ex) {
82              fail();
83          }
84  
85      }
86  }
