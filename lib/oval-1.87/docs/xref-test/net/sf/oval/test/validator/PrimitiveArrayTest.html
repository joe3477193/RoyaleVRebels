13  package net.sf.oval.test.validator;
14  
15  import java.util.List;
16  
17  import junit.framework.TestCase;
18  import net.sf.oval.ConstraintViolation;
19  import net.sf.oval.Validator;
20  import net.sf.oval.constraint.Max;
21  import net.sf.oval.constraint.MaxSize;
22  import net.sf.oval.constraint.MinSize;
23  import net.sf.oval.constraint.NotNull;
24  
25  /**
26   * @author Sebastian Thomschke
27   */
28  public class PrimitiveArrayTest extends TestCase {
29      public static class Account {
30          @MinSize(value = 1, message = "MIN_SIZE")
31          @MaxSize(value = 4, message = "MAX_SIZE")
32          @Max(value = 10, message = "MAX")
33          @NotNull(message = "NOT_NULL")
34          public int[] items = new int[] {};
35  
36      }
37  
38      public void testPrimitiveArray() {
39          final Validator validator = new Validator();
40          final Account account = new Account();
41  
42          // test min size
43          List<ConstraintViolation> violations = validator.validate(account);
44          assertEquals(1, violations.size());
45          assertEquals("MIN_SIZE", violations.get(0).getMessage());
46  
47          // test valid
48          account.items = new int[] { 1 };
49          violations = validator.validate(account);
50          assertEquals(0, violations.size());
51  
52          // test max size
53          account.items = new int[] { 1, 2, 3, 4, 5 };
54          violations = validator.validate(account);
55          assertEquals(1, violations.size());
56          assertEquals("MAX_SIZE", violations.get(0).getMessage());
57  
58          // test attribute not null
59          account.items = null;
60          violations = validator.validate(account);
61          assertEquals(1, violations.size());
62          assertEquals("NOT_NULL", violations.get(0).getMessage());
63  
64          // test elements max
65          account.items = new int[] { 1, 100 };
66          violations = validator.validate(account);
67          assertEquals(1, violations.size());
68          assertEquals("MAX", violations.get(0).getMessage());
69      }
70  }
