13  package net.sf.oval.test.constraints;
14  
15  import java.text.DateFormat;
16  import java.util.Calendar;
17  
18  import net.sf.oval.constraint.FutureCheck;
19  
20  /**
21   * @author Sebastian Thomschke
22   */
23  public class FutureTest extends AbstractContraintsTest {
24      public void testFuture() {
25          final FutureCheck check = new FutureCheck();
26          super.testCheck(check);
27          assertTrue(check.isSatisfied(null, null, null, null));
28  
29          final Calendar cal = Calendar.getInstance();
30          cal.roll(Calendar.YEAR, 1);
31          assertTrue(check.isSatisfied(null, cal, null, null));
32          assertTrue(check.isSatisfied(null, cal.getTime(), null, null));
33          assertTrue(check.isSatisfied(null, DateFormat.getDateTimeInstance().format(cal.getTime()), null, null));
34  
35          cal.roll(Calendar.YEAR, -2);
36          assertFalse(check.isSatisfied(null, cal, null, null));
37          assertFalse(check.isSatisfied(null, cal.getTime(), null, null));
38          assertFalse(check.isSatisfied(null, DateFormat.getDateTimeInstance().format(cal.getTime()), null, null));
39  
40          assertFalse(check.isSatisfied(null, "bla", null, null));
41      }
42  
43      public void testTolerance() {
44          final FutureCheck check = new FutureCheck();
45  
46          final Calendar cal = Calendar.getInstance();
47          cal.roll(Calendar.SECOND, -2);
48          assertFalse(check.isSatisfied(null, cal, null, null));
49          check.setTolerance(1500);
50          assertFalse(check.isSatisfied(null, cal, null, null));
51          check.setTolerance(5000);
52          assertTrue(check.isSatisfied(null, cal, null, null));
53      }
54  }
