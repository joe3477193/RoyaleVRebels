13  package net.sf.oval.test.validator;
14  
15  import java.util.List;
16  
17  import junit.framework.TestCase;
18  import net.sf.oval.ConstraintViolation;
19  import net.sf.oval.Validator;
20  import net.sf.oval.constraint.Assert;
21  
22  /**
23   * @author Sebastian Thomschke
24   */
25  public class AssertOGNLTest extends TestCase {
26      @net.sf.oval.constraint.Assert(expr = "_this.firstName!=null && _this.lastName!=null && (_this.firstName.length() + _this.lastName.length() > 9)", lang = "ognl", message = "C0")
27      public static class Person {
28          @Assert(expr = "_value!=null", lang = "ognl", message = "C1")
29          public String firstName;
30  
31          @Assert(expr = "_value!=null", lang = "ognl", message = "C2")
32          public String lastName;
33  
34          @Assert(expr = "_value!=null && _value.length()>0 && _value.length()<7", lang = "ognl", message = "C3")
35          public String zipCode;
36      }
37  
38      public void testOGNLExpression() {
39          final Validator validator = new Validator();
40  
41          // test not null
42          final Person p = new Person();
43          List<ConstraintViolation> violations = validator.validate(p);
44          assertTrue(violations.size() == 4);
45  
46          // test max length
47          p.firstName = "Mike";
48          p.lastName = "Mahoney";
49          p.zipCode = "1234567";
50          violations = validator.validate(p);
51          assertTrue(violations.size() == 1);
52          assertTrue(violations.get(0).getMessage().equals("C3"));
53  
54          // test not empty
55          p.zipCode = "";
56          violations = validator.validate(p);
57          assertTrue(violations.size() == 1);
58          assertTrue(violations.get(0).getMessage().equals("C3"));
59  
60          // test ok
61          p.zipCode = "wqeew";
62          violations = validator.validate(p);
63          assertTrue(violations.size() == 0);
64  
65          // test object-level constraint
66          p.firstName = "12345";
67          p.lastName = "1234";
68          violations = validator.validate(p);
69          assertTrue(violations.size() == 1);
70          assertTrue(violations.get(0).getMessage().equals("C0"));
71      }
72  }
