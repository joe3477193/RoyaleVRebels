13  package net.sf.oval.test.validator;
14  
15  import java.util.List;
16  
17  import junit.framework.TestCase;
18  import net.sf.oval.ConstraintViolation;
19  import net.sf.oval.Validator;
20  import net.sf.oval.configuration.annotation.IsInvariant;
21  import net.sf.oval.constraint.Length;
22  import net.sf.oval.constraint.NotNull;
23  import net.sf.oval.guard.SuppressOValWarnings;
24  
25  /**
26   * @author Sebastian Thomschke
27   */
28  public class InvariantMethodConstraintsValidationTest extends TestCase {
29      public static class TestEntity {
30          public String name;
31  
32          @IsInvariant
33          @NotNull(message = "NOT_NULL")
34          @Length(max = 4, message = "LENGTH")
35          public String getName() {
36              return name;
37          }
38      }
39  
40      public static class TestEntityInvalidConfig extends TestEntity {
41          /**
42           * the @NotNull annotation should lead to a "OVal API usage violation 7" warning by the ApiUsageAuditor
43           * because class is not guarded
44           */
45          public TestEntityInvalidConfig(@NotNull final String defaultValue) {
46              //
47          }
48  
49          /**
50           * the @NotNull annotation should lead to a "OVal API usage violation 2" warning by the ApiUsageAuditor
51           * because class is not guarded
52           */
53          @NotNull
54          public String doSomething(final String value) {
55              return null;
56          }
57  
58          /**
59           * the @NotNull annotation should lead to a "OVal API usage violation 5" warning by the ApiUsageAuditor
60           * because class is not guarded
61           */
62          public String doSomething(final String value, @NotNull final String defaultValue) {
63              return null;
64          }
65  
66          /**
67           * the @NotNull annotation should not lead to a "OVal API usage violation 2" warning by the ApiUsageAuditor,
68           */
69          @NotNull
70          @SuppressOValWarnings
71          public String doSomething2(final String value) {
72              return null;
73          }
74  
75          /**
76           * the @NotNull annotation should lead to a "OVal API usage violation 3" warning by the ApiUsageAuditor
77           * because @IsInvariant is missing
78           */
79          @NotNull
80          public String getSomething() {
81              return null;
82          }
83      }
84  
85      public void testMethodReturnValueConstraintValidation() {
86          final Validator validator = new Validator();
87  
88          {
89              final TestEntity t = new TestEntity();
90  
91              List<ConstraintViolation> violations = validator.validate(t);
92              assertTrue(violations.size() == 1);
93              assertTrue(violations.get(0).getMessage().equals("NOT_NULL"));
94  
95              t.name = "wqerwqer";
96              violations = validator.validate(t);
97              assertTrue(violations.size() == 1);
98              assertTrue(violations.get(0).getMessage().equals("LENGTH"));
99          }
100     }
101 }
