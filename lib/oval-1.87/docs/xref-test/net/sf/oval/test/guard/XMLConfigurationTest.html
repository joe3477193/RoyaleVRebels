67              this.managerId = managerId;

74              return managerId;
75          }

85      @SuppressWarnings("unused")
86      private static void validateUser() {
87          final ConstraintsViolatedAdapter listener = new ConstraintsViolatedAdapter();
88          TestGuardAspect.aspectOf().getGuard().addListener(listener, User.class);
89  
90          listener.clear();
91          try {
92              new User(null, null, 1);
93              fail("ConstraintViolationException expected");
94          } catch (final ConstraintsViolatedException ex) {
95              final ConstraintViolation[] violations = ex.getConstraintViolations();
96              assertEquals(2, violations.length);
97              assertEquals(User.class.getName() + "(class java.lang.String,class java.lang.String,int) parameter 0 (userId) is null", violations[0].getMessage());
98              assertEquals(User.class.getName() + "(class java.lang.String,class java.lang.String,int) parameter 1 (managerId) is null", violations[1]
99                  .getMessage());
100         }
101 
102         listener.clear();
103         try {
104             final User user = new User("12345678", "12345678", 1);
105             user.setManagerId(null);
106             fail("ConstraintViolationException expected");
107         } catch (final ConstraintsViolatedException ex) {
108             final ConstraintViolation[] violations = ex.getConstraintViolations();
109             assertEquals(1, violations.length);
110             assertEquals(User.class.getName() + ".setManagerId(class java.lang.String) parameter 0 (managerId) is null", violations[0].getMessage());
111         }
112 
113         listener.clear();
114         try {
115             final User user = new User();
116             user.getManagerId();
117             fail("ConstraintViolationException expected");
118         } catch (final ConstraintsViolatedException ex) {
119             final ConstraintViolation[] violations = ex.getConstraintViolations();
120             assertEquals(1, violations.length);
121             assertEquals(User.class.getName() + ".getManagerId() is null", violations[0].getMessage());
122         }
123     }

135     public void testImportedFile() {
136         try {
137             final XMLConfigurer x = new XMLConfigurer();
138             x.fromXML(XMLConfigurationTest.class.getResourceAsStream("XMLConfigurationTest.xml"));
139 
140             final Guard guard = new Guard(x);
141             guard.setInvariantsEnabled(false);
142             TestGuardAspect.aspectOf().setGuard(guard);
143 
144             validateUser();
145         } catch (final ValidationFailedException ex) {
146             ex.getCause().printStackTrace();
147             throw ex;
148         }
149     }

151     public void testSerializedObjectConfiguration() {
152         final XMLConfigurer x = new XMLConfigurer();
153 
154         /*
155          * define a configuration
156          */
157         final Set<ConstraintSetConfiguration> constraintSetsConfig = new HashSet<ConstraintSetConfiguration>();
158         {
159             final ConstraintSetConfiguration csf = new ConstraintSetConfiguration();
160             constraintSetsConfig.add(csf);
161 
162             csf.id = "user.userid";
163             csf.checks = new ArrayList<Check>();
164             final NotNullCheck nnc = new NotNullCheck();
165             nnc.setMessage("{context} is null");
166             csf.checks.add(nnc);
167             final MatchPatternCheck rec = new MatchPatternCheck();
168             rec.setPattern(Pattern.compile("^[a-z0-9]{8}$", 0));
169             rec.setMessage("{context} does not match the pattern {pattern}");
170             csf.checks.add(rec);
171         }
172 
173         final Set<ClassConfiguration> classConfigs = new HashSet<ClassConfiguration>();
174         {
175             final ClassConfiguration cf = new ClassConfiguration();
176             classConfigs.add(cf);
177             cf.type = User.class;
178 
179             cf.fieldConfigurations = new HashSet<FieldConfiguration>();
180             {
181                 final FieldConfiguration fc = new FieldConfiguration();
182                 cf.fieldConfigurations.add(fc);
183 
184                 fc.name = "firstName";
185                 fc.checks = new ArrayList<Check>();
186                 final LengthCheck lc = new LengthCheck();
187                 lc.setMessage("{context} is not between {min} and {max} characters long");
188                 lc.setMax(3);
189                 fc.checks.add(lc);
190             }
191             {
192                 final FieldConfiguration fc = new FieldConfiguration();
193                 cf.fieldConfigurations.add(fc);
194 
195                 fc.name = "lastName";
196                 fc.checks = new ArrayList<Check>();
197                 final LengthCheck lc = new LengthCheck();
198                 lc.setMessage("{context} is not between {min} and {max} characters long");
199                 lc.setMax(5);
200                 fc.checks.add(lc);
201             }
202             {
203                 final FieldConfiguration fc = new FieldConfiguration();
204                 fc.overwrite = Boolean.TRUE;
205                 cf.fieldConfigurations.add(fc);
206 
207                 fc.name = "userId";
208                 fc.checks = new ArrayList<Check>();
209                 final AssertConstraintSetCheck acsc = new AssertConstraintSetCheck();
210                 acsc.setId("user.userid");
211                 fc.checks.add(acsc);
212             }
213 
214             cf.constructorConfigurations = new HashSet<ConstructorConfiguration>();
215             {
216                 final ConstructorConfiguration cc = new ConstructorConfiguration();
217                 cf.constructorConfigurations.add(cc);
218                 cc.parameterConfigurations = new ArrayList<ParameterConfiguration>();
219 
220                 final AssertConstraintSetCheck acsc = new AssertConstraintSetCheck();
221                 acsc.setId("user.userid");
222 
223                 final ParameterConfiguration pc1 = new ParameterConfiguration();
224                 pc1.type = String.class;
225                 pc1.checks = new ArrayList<Check>();
226                 pc1.checks.add(acsc);
227                 cc.parameterConfigurations.add(pc1);
228                 final ParameterConfiguration pc2 = new ParameterConfiguration();
229                 pc2.type = String.class;
230                 pc2.checks = new ArrayList<Check>();
231                 pc2.checks.add(acsc);
232                 cc.parameterConfigurations.add(pc2);
233                 final ParameterConfiguration pc3 = new ParameterConfiguration();
234                 pc3.type = int.class;
235                 cc.parameterConfigurations.add(pc3);
236             }
237 
238             cf.methodConfigurations = new HashSet<MethodConfiguration>();
239             {
240                 final AssertConstraintSetCheck acsc = new AssertConstraintSetCheck();
241                 acsc.setId("user.userid");
242 
243                 MethodConfiguration mc = new MethodConfiguration();
244                 cf.methodConfigurations.add(mc);
245                 mc.name = "getManagerId";
246                 mc.returnValueConfiguration = new MethodReturnValueConfiguration();
247                 mc.returnValueConfiguration.checks = new ArrayList<Check>();
248                 mc.returnValueConfiguration.checks.add(acsc);
249 
250                 mc = new MethodConfiguration();
251                 cf.methodConfigurations.add(mc);
252                 mc.name = "setManagerId";
253                 mc.parameterConfigurations = new ArrayList<ParameterConfiguration>();
254                 final ParameterConfiguration pc1 = new ParameterConfiguration();
255                 pc1.type = String.class;
256                 pc1.checks = new ArrayList<Check>();
257                 pc1.checks.add(acsc);
258                 mc.parameterConfigurations.add(pc1);
259             }
260         }
261 
262         x.getPojoConfigurer().setClassConfigurations(classConfigs);
263         x.getPojoConfigurer().setConstraintSetConfigurations(constraintSetsConfig);
264 
265         /*
266          * serialize the configuration to XML
267          */
268         final String xmlConfig = x.toXML();
269 
270         /*
271          * deserialize the configuration from XML
272          */
273         x.fromXML(xmlConfig);
274 
275         final Guard guard = new Guard(x);
276         guard.setInvariantsEnabled(false);
277         TestGuardAspect.aspectOf().setGuard(guard);
278 
279         validateUser();
280     }
