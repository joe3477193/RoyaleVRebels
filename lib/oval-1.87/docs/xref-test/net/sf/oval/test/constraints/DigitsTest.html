13  package net.sf.oval.test.constraints;
14  
15  import java.math.BigDecimal;
16  
17  import net.sf.oval.constraint.DigitsCheck;
18  
19  /**
20   * @author Sebastian Thomschke
21   */
22  public class DigitsTest extends AbstractContraintsTest {
23      public void testDigits() {
24          final DigitsCheck check = new DigitsCheck();
25          super.testCheck(check);
26          check.setMaxFraction(2);
27          check.setMaxInteger(2);
28  
29          assertTrue(check.isSatisfied(null, null, null, null));
30  
31          assertTrue(check.isSatisfied(null, 0, null, null));
32          assertTrue(check.isSatisfied(null, 0.0, null, null));
33          assertTrue(check.isSatisfied(null, 0.00, null, null));
34          assertTrue(check.isSatisfied(null, 0.000, null, null)); // returns true since it equals 0.0d
35          assertFalse(check.isSatisfied(null, 0.001, null, null));
36          assertTrue(check.isSatisfied(null, 000.0, null, null)); // returns true since it equals 0.0d
37          assertTrue(check.isSatisfied(null, 12, null, null));
38          assertFalse(check.isSatisfied(null, 123, null, null));
39          assertTrue(check.isSatisfied(null, 12.12, null, null));
40          assertFalse(check.isSatisfied(null, 12.123, null, null));
41          assertFalse(check.isSatisfied(null, 123.12, null, null));
42  
43          assertTrue(check.isSatisfied(null, "0", null, null));
44          assertTrue(check.isSatisfied(null, "0.0", null, null));
45          assertTrue(check.isSatisfied(null, "0.00", null, null));
46          assertFalse(check.isSatisfied(null, "0.000", null, null)); // returns false since BigDecimal keeps the fraction
47          assertFalse(check.isSatisfied(null, "0.001", null, null));
48          assertTrue(check.isSatisfied(null, "000.0", null, null)); // TODO returns true since it is automatically converted into a BigDecimal("0.0") => should return false in case of pure strings?
49          assertTrue(check.isSatisfied(null, "12", null, null));
50          assertFalse(check.isSatisfied(null, "123", null, null));
51          assertTrue(check.isSatisfied(null, "12.12", null, null));
52          assertFalse(check.isSatisfied(null, "12.123", null, null));
53          assertFalse(check.isSatisfied(null, "123.12", null, null));
54  
55          assertTrue(check.isSatisfied(null, new BigDecimal("0"), null, null));
56          assertTrue(check.isSatisfied(null, new BigDecimal("0.0"), null, null));
57          assertTrue(check.isSatisfied(null, new BigDecimal("0.00"), null, null));
58          assertFalse(check.isSatisfied(null, new BigDecimal("0.000"), null, null)); // TODO returns false since BigDecimal keeps the fraction => should this return true for BigDecimals ?
59          assertFalse(check.isSatisfied(null, new BigDecimal("0.001"), null, null));
60          assertTrue(check.isSatisfied(null, new BigDecimal("000.0"), null, null)); // returns true since it is automatically converted into a BigDecimal("0.0")
61          assertTrue(check.isSatisfied(null, new BigDecimal("12"), null, null));
62          assertFalse(check.isSatisfied(null, new BigDecimal("123"), null, null));
63          assertTrue(check.isSatisfied(null, new BigDecimal("12.12"), null, null));
64          assertFalse(check.isSatisfied(null, new BigDecimal("12.123"), null, null));
65          assertFalse(check.isSatisfied(null, new BigDecimal("123.12"), null, null));
66  
67          check.setMaxInteger(13);
68          check.setMaxFraction(13);
69          assertTrue(check.isSatisfied(null, 1234567890123L, null, null));
70          assertFalse(check.isSatisfied(null, 12345678901234L, null, null));
71          assertTrue(check.isSatisfied(null, "1234567890123", null, null));
72          assertFalse(check.isSatisfied(null, "12345678901234", null, null));
73          assertTrue(check.isSatisfied(null, new BigDecimal("1234567890123"), null, null));
74          assertFalse(check.isSatisfied(null, new BigDecimal("12345678901234"), null, null));
75          assertTrue(check.isSatisfied(null, new BigDecimal("1234567890123.1234567890123"), null, null));
76          assertFalse(check.isSatisfied(null, new BigDecimal("12345678901234.12345678901234"), null, null));
77      }
78  }
