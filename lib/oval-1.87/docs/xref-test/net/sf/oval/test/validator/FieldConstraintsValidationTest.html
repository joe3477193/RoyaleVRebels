13  package net.sf.oval.test.validator;
14  
15  import java.util.List;
16  
17  import junit.framework.TestCase;
18  import net.sf.oval.ConstraintViolation;
19  import net.sf.oval.Validator;
20  import net.sf.oval.constraint.Length;
21  import net.sf.oval.constraint.MatchPattern;
22  import net.sf.oval.constraint.NotEmpty;
23  import net.sf.oval.constraint.NotNull;
24  
25  /**
26   * @author Sebastian Thomschke
27   */
28  public class FieldConstraintsValidationTest extends TestCase {
29      protected static class Person {
30          @NotNull
31          public String firstName;
32  
33          @NotNull
34          public String lastName;
35  
36          @NotNull
37          @Length(max = 6, message = "LENGTH")
38          @NotEmpty(message = "NOT_EMPTY")
39          @MatchPattern(pattern = PATTERN_ZIP_CODE, message = "MATCH_PATTERN")
40          public String zipCode;
41      }
42  
43      private static final String PATTERN_ZIP_CODE = "^[0-9]*$";
44  
45      public void testFieldValidation() {
46          final Validator validator = new Validator();
47  
48          // test @NotNull
49          final Person p = new Person();
50          List<ConstraintViolation> violations = validator.validate(p);
51          assertTrue(violations.size() == 3);
52  
53          // test @Length(max=)
54          p.firstName = "Mike";
55          p.lastName = "Mahoney";
56          p.zipCode = "1234567";
57          violations = validator.validate(p);
58          assertTrue(violations.size() == 1);
59          assertTrue(violations.get(0).getMessage().equals("LENGTH"));
60  
61          // test @NotEmpty
62          p.zipCode = "";
63          violations = validator.validate(p);
64          assertTrue(violations.size() == 1);
65          assertTrue(violations.get(0).getMessage().equals("NOT_EMPTY"));
66  
67          // test @RegEx
68          p.zipCode = "dffd34";
69          violations = validator.validate(p);
70          assertTrue(violations.size() == 1);
71          assertTrue(violations.get(0).getMessage().equals("MATCH_PATTERN"));
72      }
73  }
