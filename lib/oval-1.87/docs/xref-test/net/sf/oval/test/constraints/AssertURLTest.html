13  package net.sf.oval.test.constraints;
14  
15  import net.sf.oval.constraint.AssertURLCheck;
16  import net.sf.oval.constraint.AssertURLCheck.URIScheme;
17  
18  /**
19   * @author Makkari
20   * @author Sebastian Thomschke
21   */
22  public class AssertURLTest extends AbstractContraintsTest {
23      private static final URIScheme[] PERMITTED_SCHEMES = { URIScheme.FTP, URIScheme.HTTP, URIScheme.HTTPS };
24  
25      public void testAssertURL() {
26          final AssertURLCheck check = new AssertURLCheck();
27          super.testCheck(check);
28          assertNull(check.getPermittedURISchemes());
29  
30          check.setPermittedURISchemes(AssertURLTest.PERMITTED_SCHEMES);
31          final URIScheme[] actualPermittedSchemes = check.getPermittedURISchemes();
32          assertNotNull(actualPermittedSchemes);
33          assertEquals(AssertURLTest.PERMITTED_SCHEMES.length, actualPermittedSchemes.length);
34          for (int n = 0; n < AssertURLTest.PERMITTED_SCHEMES.length; n++) {
35              assertEquals(AssertURLTest.PERMITTED_SCHEMES[n], actualPermittedSchemes[n]);
36              assertEquals(AssertURLTest.PERMITTED_SCHEMES[n].toString(), actualPermittedSchemes[n].getScheme());
37          }
38  
39          assertFalse(check.isConnect());
40          check.setConnect(true);
41          assertTrue(check.isConnect());
42  
43          check.setConnect(false);
44          assertTrue(check.isSatisfied(this, null, null, validator));
45          assertFalse(check.isSatisfied(this, "http", null, validator));
46          assertFalse(check.isSatisfied(this, "https", null, validator));
47          assertFalse(check.isSatisfied(this, "ftp", null, validator));
48          assertTrue(check.isSatisfied(this, "http://www.google.com", null, validator));
49          assertTrue(check.isSatisfied(this, "https://www.google.com", null, validator));
50          assertTrue(check.isSatisfied(this, "httPs://www.google.com", null, validator));
51          assertTrue(check.isSatisfied(this, "ftp://ftp.is.co.za/rfc/rfc1808.txt", null, validator));
52          assertFalse(check.isSatisfied(this, "ptth://www.google.com", null, validator));
53          assertFalse(check.isSatisfied(this, "http://www.g[oogle.com", null, validator));
54  
55          check.setConnect(true);
56          assertTrue(check.isSatisfied(this, null, null, validator));
57          assertFalse(check.isSatisfied(this, "http", null, validator));
58          assertFalse(check.isSatisfied(this, "https", null, validator));
59          assertFalse(check.isSatisfied(this, "ftp", null, validator));
60          assertTrue(check.isSatisfied(this, "http://www.google.com", null, validator));
61          //assertTrue(check.isSatisfied(this, "https://www.verisign.com/site-map/index.html", null, validator));
62          assertFalse(check.isSatisfied(this, "http://127.0.0.1:34343", null, validator));
63          assertTrue(check.isSatisfied(this, "ftp://ftp.debian.org/debian/README.html", null, validator));
64          assertFalse(check.isSatisfied(this, "ftp://ftp.debian.org/debian/foo.html", null, validator));
65  
66          check.setPermittedURISchemes(null);
67      }
68  }
