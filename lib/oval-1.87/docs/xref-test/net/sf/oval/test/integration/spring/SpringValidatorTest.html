13  package net.sf.oval.test.integration.spring;
14  
15  import junit.framework.TestCase;
16  import net.sf.oval.Validator;
17  import net.sf.oval.constraint.NotNegative;
18  import net.sf.oval.constraint.NotNull;
19  import net.sf.oval.integration.spring.SpringValidator;
20  
21  import org.springframework.validation.BindException;
22  
23  /**
24   * @author Sebastian Thomschke
25   */
26  public class SpringValidatorTest extends TestCase {
27      public static class Entity {
28          @NotNull(errorCode = "E1", message = "M1")
29          protected String name;
30  
31          @NotNegative(errorCode = "E2", message = "M2")
32          protected int age;
33  
34          public int getAge() {
35              return age;
36          }
37  
38          public String getName() {
39              return name;
40          }
41      }
42  
43      public void testSpringValidator() {
44          final SpringValidator v = new SpringValidator(new Validator());
45          final Entity e = new Entity();
46          {
47              e.name = null;
48              e.age = -1;
49              final BindException errors = new BindException(e, e.getClass().getName());
50              v.validate(e, errors);
51              assertEquals(2, errors.getErrorCount());
52              assertEquals(2, errors.getFieldErrorCount());
53              assertEquals(null, errors.getFieldError("name").getRejectedValue());
54              assertTrue(errors.getFieldError("name").getCodes()[0].startsWith("E1"));
55              assertEquals(-1, errors.getFieldError("age").getRejectedValue());
56              assertTrue(errors.getFieldError("age").getCodes()[0].startsWith("E2"));
57          }
58          {
59              final BindException errors = new BindException(e, e.getClass().getName());
60              e.name = "";
61              e.age = -1;
62              v.validate(e, errors);
63              assertEquals(1, errors.getErrorCount());
64              assertEquals(1, errors.getFieldErrorCount());
65              assertEquals(-1, errors.getFieldError("age").getRejectedValue());
66              assertTrue(errors.getFieldError("age").getCodes()[0].startsWith("E2"));
67          }
68          {
69              final BindException errors = new BindException(e, e.getClass().getName());
70              e.name = null;
71              e.age = 0;
72              v.validate(e, errors);
73              assertEquals(1, errors.getErrorCount());
74              assertEquals(1, errors.getFieldErrorCount());
75              assertEquals(null, errors.getFieldError("name").getRejectedValue());
76              assertTrue(errors.getFieldError("name").getCodes()[0].startsWith("E1"));
77          }
78          {
79              final BindException errors = new BindException(e, e.getClass().getName());
80              e.name = "";
81              e.age = 0;
82              v.validate(e, errors);
83              assertEquals(0, errors.getErrorCount());
84              assertEquals(0, errors.getFieldErrorCount());
85          }
86      }
87  }
