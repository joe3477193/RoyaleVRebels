13  package net.sf.oval.test.constraints;
14  
15  import java.text.DateFormat;
16  import java.util.Calendar;
17  import java.util.Date;
18  
19  import net.sf.oval.constraint.DateRangeCheck;
20  
21  /**
22   * @author Sebastian Thomschke
23   */
24  public class DateRangeTest extends AbstractContraintsTest {
25      public void testDateRange() {
26          final DateRangeCheck check = new DateRangeCheck();
27          super.testCheck(check);
28          assertTrue(check.isSatisfied(null, null, null, null));
29          check.setFormat("yyyy-MM-dd HH:mm:ss");
30  
31          check.setMin("2000-03-03 09:09:09");
32          assertFalse(check.isSatisfied(null, "2000-03-03 09:09:08", null, null));
33          assertTrue(check.isSatisfied(null, "2000-03-03 09:09:09", null, null));
34          assertTrue(check.isSatisfied(null, "2000-03-03 09:09:10", null, null));
35          assertTrue(check.isSatisfied(null, "2900-03-03 09:09:09", null, null));
36  
37          check.setMax("2080-08-08 10:10:10");
38          assertTrue(check.isSatisfied(null, "2080-08-08 10:10:09", null, null));
39          assertTrue(check.isSatisfied(null, "2080-08-08 10:10:10", null, null));
40          assertFalse(check.isSatisfied(null, "2080-08-08 10:10:11", null, null));
41  
42          final Calendar cal = Calendar.getInstance();
43          assertTrue(check.isSatisfied(null, cal, null, null));
44          assertTrue(check.isSatisfied(null, cal.getTime(), null, null));
45          assertTrue(check.isSatisfied(null, DateFormat.getDateTimeInstance().format(cal.getTime()), null, null));
46  
47          cal.add(Calendar.YEAR, -100);
48          assertFalse(check.isSatisfied(null, cal, null, null));
49          assertFalse(check.isSatisfied(null, cal.getTime(), null, null));
50          assertFalse(check.isSatisfied(null, DateFormat.getDateTimeInstance().format(cal.getTime()), null, null));
51  
52          cal.add(Calendar.YEAR, 200);
53          assertFalse(check.isSatisfied(null, cal, null, null));
54          assertFalse(check.isSatisfied(null, cal.getTime(), null, null));
55          assertFalse(check.isSatisfied(null, DateFormat.getDateTimeInstance().format(cal.getTime()), null, null));
56  
57          assertFalse(check.isSatisfied(null, "bla", null, null));
58      }
59  
60      public void testLiterals() {
61          final DateRangeCheck check = new DateRangeCheck();
62          check.setFormat("yyyy-MM-dd HH:mm:ss");
63  
64          check.setMin("now");
65          assertTrue(check.isSatisfied(null, new Date(), null, null));
66          check.setMin("today");
67          assertTrue(check.isSatisfied(null, new Date(), null, null));
68          check.setMin("tomorrow");
69          assertFalse(check.isSatisfied(null, new Date(), null, null));
70          check.setMin("yesterday");
71          assertTrue(check.isSatisfied(null, new Date(), null, null));
72          check.setMin("");
73  
74          check.setMax("now");
75          assertFalse(check.isSatisfied(null, new Date(System.currentTimeMillis() + 4000), null, null));
76          assertTrue(check.isSatisfied(null, new Date(System.currentTimeMillis() - 4000), null, null));
77          check.setMax("today");
78          assertTrue(check.isSatisfied(null, new Date(), null, null));
79          assertTrue(check.isSatisfied(null, "2000-03-03 09:09:10", null, null));
80          check.setMax("tomorrow");
81          assertTrue(check.isSatisfied(null, new Date(), null, null));
82          assertTrue(check.isSatisfied(null, "2000-03-03 09:09:10", null, null));
83          check.setMax("yesterday");
84          assertFalse(check.isSatisfied(null, new Date(), null, null));
85          check.setMax("");
86      }
87  
88      public void testTolerance() {
89          final DateRangeCheck check = new DateRangeCheck();
90          check.setMax("now");
91          check.setTolerance(4000);
92          assertTrue(check.isSatisfied(null, new Date(System.currentTimeMillis() + 2000), null, null));
93          check.setTolerance(0);
94          assertFalse(check.isSatisfied(null, new Date(System.currentTimeMillis() + 2000), null, null));
95          check.setMax("");
96  
97          check.setMax("now");
98          check.setTolerance(4000);
99          assertTrue(check.isSatisfied(null, new Date(System.currentTimeMillis() + 2000), null, null));
100         check.setTolerance(0);
101         assertFalse(check.isSatisfied(null, new Date(System.currentTimeMillis() + 2000), null, null));
102         check.setMax("");
103     }
