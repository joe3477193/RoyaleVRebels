13  package net.sf.oval.test.validator;
14  
15  import java.lang.annotation.Annotation;
16  import java.util.Enumeration;
17  import java.util.ResourceBundle;
18  
19  import junit.framework.TestCase;
20  import net.sf.oval.Check;
21  import net.sf.oval.internal.util.ReflectionUtils;
22  
23  /**
24   * @author Sebastian Thomschke
25   */
26  public class ConstraintViolationMessagesTest extends TestCase {
27  
28      public void testMessages() throws ClassNotFoundException, InstantiationException, IllegalAccessException {
29          ResourceBundle bundle = ResourceBundle.getBundle("net.sf.oval.Messages");
30          for (Enumeration<String> en = bundle.getKeys(); en.hasMoreElements();) {
31              String key = en.nextElement();
32              if (key.endsWith(".violated")) {
33                  String className = key.substring(0, key.length() - 9);
34  
35                  @SuppressWarnings("unchecked")
36                  Class<Annotation> annotationClass = (Class<Annotation>) Class.forName(className);
37  
38                  // check that the default message defined on the annotation is the same as the key read from the bundle
39                  String annotationMessage = (String) ReflectionUtils.getMethod(annotationClass, "message").getDefaultValue();
40                  assertEquals(key, annotationMessage);
41                  String annotationErrorCode = (String) ReflectionUtils.getMethod(annotationClass, "errorCode").getDefaultValue();
42                  assertEquals(className, annotationErrorCode);
43  
44                  // check that the message key returned by the check instance is the same as the key read from the bundle
45                  Check check = (Check) Class.forName(className + "Check").newInstance();
46                  assertEquals(key, check.getMessage());
47              } else if (key.endsWith(".parameter")) {
48                  String className = key.substring(0, key.length() - 10);
49                  Class.forName(className);
50              }
51          }
52      }
53  }
